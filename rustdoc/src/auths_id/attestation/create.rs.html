19/// Current attestation version - includes org fields in signed envelope
20pub const ATTESTATION_VERSION: u32 = 1;

22/// Maximum allowed clock drift at creation time (seconds)
23const MAX_CREATION_SKEW_SECS: i64 = 5 * 60;

25/// NEW: Data structure specifically for canonicalizing revocation statements.
26/// Excludes fields not relevant to the revocation itself (device_pk, payload, expires_at).
27#[derive(Serialize, Debug)] // Added Debug
28pub struct CanonicalRevocationData<'a> {
29    pub version: u32,
30    pub rid: &'a str,
31    pub issuer: &'a IdentityDID,
32    pub subject: &'a DeviceDID,
33    pub timestamp: &'a Option<DateTime<Utc>>,
34    pub revoked_at: &'a Option<DateTime<Utc>>, // Should always be Some(...)
35    pub note: &'a Option<String>,
36}

38/// Creates a signed attestation by signing internally using the provided SecureSigner.
39///
40/// This function constructs the canonical attestation data, signs it using the signer
41/// for both identity and device (if device_alias is provided), and returns the complete
42/// attestation with embedded signatures.
43///
44/// # Arguments
45/// * `rid` - Resource identifier for this attestation
46/// * `identity_did` - The identity DID (e.g., "did:keri:...") issuing the attestation
47/// * `device_did` - The device DID being attested
48/// * `device_public_key` - The 32-byte Ed25519 public key of the device
49/// * `payload` - Optional JSON payload for the attestation
50/// * `meta` - Attestation metadata (timestamp, expiry, notes)
51/// * `signer` - SecureSigner implementation for signing operations
52/// * `passphrase_provider` - Provider for obtaining passphrases during signing
53/// * `identity_alias` - Optional alias of the identity key in the keychain (None = device-only signing)
54/// * `device_alias` - Optional alias of the device key (None means no device signature)
55/// * `capabilities` - Capabilities to grant (included in the signed envelope)
56/// * `role` - Optional org role (e.g., "admin", "member") included in the signed envelope
57/// * `delegated_by` - Optional DID of the delegator included in the signed envelope
58#[allow(clippy::too_many_arguments)]
59pub fn create_signed_attestation(
60    now: DateTime<Utc>,
61    rid: &str,
62    identity_did: &IdentityDID,
63    device_did: &DeviceDID,
64    device_public_key: &[u8],
65    payload: Option<Value>,
66    meta: &AttestationMetadata,
67    signer: &dyn SecureSigner,
68    passphrase_provider: &dyn PassphraseProvider,
69    identity_alias: Option<&KeyAlias>,
70    device_alias: Option<&KeyAlias>,
71    capabilities: Vec<Capability>,
72    role: Option<Role>,
73    delegated_by: Option<IdentityDID>,
74) -> Result<Attestation, AttestationError> {
75    if device_public_key.len() != ED25519_PUBLIC_KEY_LEN {
76        return Err(AttestationError::InvalidInput(format!(
77            "Device public key length must be {}",
78            ED25519_PUBLIC_KEY_LEN
79        )));
80    }
81
82    // Validate timestamp is not too far from current time (clock drift protection)
83    if let Some(ts) = meta.timestamp {
84        let drift = (now - ts).num_seconds().abs();
85        if drift > MAX_CREATION_SKEW_SECS {
86            return Err(AttestationError::InvalidInput(format!(
87                "System clock drift {}s exceeds {}s limit",
88                drift, MAX_CREATION_SKEW_SECS
89            )));
90        }
91    }
92
93    // Construct the canonical data to be signed
94    let data_to_canonicalize = CanonicalAttestationData {
95        version: ATTESTATION_VERSION,
96        rid,
97        issuer: identity_did,
98        subject: device_did,
99        device_public_key,
100        payload: &payload,
101        timestamp: &meta.timestamp,
102        expires_at: &meta.expires_at,
103        revoked_at: &None,
104        note: &meta.note,
105        // Org fields included in signed envelope
106        role: role.as_ref().map(|r| r.as_str()),
107        capabilities: if capabilities.is_empty() {
108            None
109        } else {
110            Some(&capabilities)
111        },
112        delegated_by: delegated_by.as_ref(),
113        signer_type: None,
114    };
115
116    // Canonicalize the attestation data
117    let message_to_sign = canonicalize_attestation_data(&data_to_canonicalize)?;
118
119    // Sign with the identity key (if alias provided)
120    let identity_signature = if let Some(alias) = identity_alias {
121        debug!("Signing attestation with identity alias '{}'", alias);
122        let sig = signer
123            .sign_with_alias(alias, passphrase_provider, &message_to_sign)
124            .map_err(|e| {
125                AttestationError::SigningError(format!(
126                    "Failed to sign with identity key '{}': {}",
127                    alias, e
128                ))
129            })?;
130        debug!("Identity signature obtained successfully");
131        Ed25519Signature::try_from_slice(&sig)
132            .map_err(|e| AttestationError::SigningError(e.to_string()))?
133    } else {
134        debug!("No identity alias provided, skipping identity signature (device-only attestation)");
135        Ed25519Signature::empty()
136    };
137
138    // Sign with the device key if alias provided
139    let device_signature = if let Some(alias) = device_alias {
140        debug!("Signing attestation with device alias '{}'", alias);
141        let sig = signer
142            .sign_with_alias(alias, passphrase_provider, &message_to_sign)
143            .map_err(|e| {
144                AttestationError::SigningError(format!(
145                    "Failed to sign with device key '{}': {}",
146                    alias, e
147                ))
148            })?;
149        debug!("Device signature obtained successfully");
150        Ed25519Signature::try_from_slice(&sig)
151            .map_err(|e| AttestationError::SigningError(e.to_string()))?
152    } else {
153        debug!("No device alias provided, skipping device signature");
154        Ed25519Signature::empty()
155    };
156
157    // Construct final attestation
158    Ok(Attestation {
159        version: ATTESTATION_VERSION,
160        subject: device_did.clone(),
161        issuer: identity_did.clone(),
162        rid: ResourceId::new(rid),
163        payload: payload.clone(),
164        timestamp: meta.timestamp,
165        expires_at: meta.expires_at,
166        revoked_at: None,
167        note: meta.note.clone(),
168        device_public_key: Ed25519PublicKey::try_from_slice(device_public_key)
169            .map_err(|e| AttestationError::InvalidInput(e.to_string()))?,
170        identity_signature,
171        device_signature,
172        role,
173        capabilities,
174        delegated_by,
175        signer_type: None,
176        environment_claim: None,
177    })
178}

180/// Generates the canonical byte representation specifically for revocation data.
181pub fn canonicalize_revocation_data(
182    data: &CanonicalRevocationData,
183) -> Result<Vec<u8>, AttestationError> {
184    let canonical_json_string = json_canon::to_string(data).map_err(|e| {
185        AttestationError::SerializationError(format!(
186            "Failed to create canonical JSON for revocation: {}",
187            e
188        ))
189    })?;
190    debug!(
191        "Generated canonical data (revocation): {}",
192        canonical_json_string
193    );
194    Ok(canonical_json_string.into_bytes())
195}
