1use auths_verifier::core::Attestation;
2use auths_verifier::types::DeviceDID;
3
4use std::collections::BTreeMap;
5
6/// A group of attestations indexed by device DID (subject).
7#[derive(Debug)]
8pub struct AttestationGroup {
9    /// Map of device DID string (subject) → list of attestations
10    pub by_device: BTreeMap<String, Vec<Attestation>>,
11}
12
13impl AttestationGroup {
14    /// Groups a list of attestations by their `subject` field (device DID).
15    pub fn from_list(attestations: Vec<Attestation>) -> Self {
16        let mut map: BTreeMap<String, Vec<Attestation>> = BTreeMap::new();
17
18        for att in attestations {
19            // Use the new field name 'subject'
20            let key = att.subject.as_str().to_owned();
21            map.entry(key).or_default().push(att);
22        }
23
24        Self { by_device: map }
25    }
26
27    /// Returns the number of distinct devices (subjects) found.
28    pub fn device_count(&self) -> usize {
29        self.by_device.len()
30    }
31
32    /// Flattens into all attestations in sorted key order.
33    pub fn all(&self) -> Vec<&Attestation> {
34        self.by_device.values().flat_map(|v| v.iter()).collect()
35    }
36
37    /// Returns the attestations for a specific device DID string (subject).
38    pub fn get(&self, device_did_str: &str) -> Option<&Vec<Attestation>> {
39        self.by_device.get(device_did_str)
40    }
41
42    /// Returns the most recent attestation for the given device DID (subject).
43    pub fn latest(&self, device_did: &DeviceDID) -> Option<&Attestation> {
44        self.by_device
45            .get(device_did.as_str())
46            .and_then(|list| list.last())
47    }
48}
