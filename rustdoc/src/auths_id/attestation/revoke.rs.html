12/// Revocation version - stays at v1 since revocations don't need org fields
13pub const REVOCATION_VERSION: u32 = 1;
14
15/// Creates a signed revocation attestation using the provided SecureSigner.
16///
17/// This function constructs the canonical revocation data, signs it using the
18/// identity key via the signer, and returns the complete revocation attestation.
19///
20/// # Arguments
21/// * `rid` - Resource identifier for the attestation being revoked
22/// * `identity_did` - The identity DID (e.g., "did:keri:...") issuing the revocation
23/// * `device_did` - The device DID being revoked
24/// * `device_public_key` - The device's Ed25519 public key bytes (from existing attestation)
25/// * `note` - Optional note explaining the revocation reason
26/// * `payload_arg` - Optional JSON payload (usually None for revocations)
27/// * `timestamp_arg` - Timestamp of the revocation
28/// * `signer` - SecureSigner implementation for signing operations
29/// * `passphrase_provider` - Provider for obtaining passphrases during signing
30/// * `identity_alias` - Alias of the identity key in the keychain
31#[allow(clippy::too_many_arguments)]
32pub fn create_signed_revocation(
33    rid: &str,
34    identity_did: &IdentityDID,
35    device_did: &DeviceDID,
36    device_public_key: &[u8],
37    note: Option<String>,
38    payload_arg: Option<Value>,
39    timestamp_arg: DateTime<Utc>,
40    signer: &dyn SecureSigner,
41    passphrase_provider: &dyn PassphraseProvider,
42    identity_alias: &KeyAlias,
43) -> Result<Attestation, AttestationError> {
44    warn!("Creating revocation for device {}", device_did);
45
46    // 1. Construct the revocation-specific canonical data
47    let revoked_at_value = Some(timestamp_arg);
48    let data_to_canonicalize_revocation = CanonicalRevocationData {
49        version: REVOCATION_VERSION,
50        rid,
51        issuer: identity_did,
52        subject: device_did,
53        timestamp: &Some(timestamp_arg),
54        revoked_at: &revoked_at_value,
55        note: &note,
56    };
57
58    // 2. Canonicalize the revocation data
59    let canonical_bytes = canonicalize_revocation_data(&data_to_canonicalize_revocation)?;
60    debug!(
61        "Canonical revocation data: {}",
62        String::from_utf8_lossy(&canonical_bytes)
63    );
64
65    // 3. Sign with the identity key
66    debug!(
67        "Signing revocation with identity alias '{}'",
68        identity_alias
69    );
70    let identity_sig_bytes = signer
71        .sign_with_alias(identity_alias, passphrase_provider, &canonical_bytes)
72        .map_err(|e| {
73            AttestationError::SigningError(format!(
74                "Failed to sign revocation with identity key '{}': {}",
75                identity_alias, e
76            ))
77        })?;
78    let identity_signature = Ed25519Signature::try_from_slice(&identity_sig_bytes)
79        .map_err(|e| AttestationError::SigningError(e.to_string()))?;
80    debug!("Revocation signature obtained successfully");
81
82    // 4. Return the final revocation attestation object
83    Ok(Attestation {
84        version: REVOCATION_VERSION,
85        subject: device_did.clone(),
86        issuer: identity_did.clone(),
87        rid: ResourceId::new(rid),
88        payload: payload_arg.clone(),
89        timestamp: Some(timestamp_arg),
90        expires_at: None,
91        revoked_at: Some(timestamp_arg),
92        note: note.clone(),
93        device_public_key: Ed25519PublicKey::try_from_slice(device_public_key)
94            .map_err(|e| AttestationError::InvalidInput(e.to_string()))?,
95        identity_signature,
96        device_signature: Ed25519Signature::empty(),
97        role: None,
98        capabilities: vec![],
99        delegated_by: None,
100        signer_type: None,
101        environment_claim: None,
102    })
103}
