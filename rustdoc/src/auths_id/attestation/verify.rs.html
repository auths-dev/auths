4use chrono::{DateTime, Duration, Utc};
5use log::debug;
6use ring::signature::{ED25519, UnparsedPublicKey};

8/// Maximum allowed time skew for attestation timestamps.
9const MAX_SKEW_SECS: i64 = 5 * 60;
10
11/// Verifies an attestation's signatures, revocation status, expiry, and timestamp validity.
12///
13/// **Timestamp threat model:** Past timestamps are accepted by default because attestations
14/// stored in Git are routinely verified days or months after issuance. Only future timestamps
15/// (beyond `MAX_SKEW_SECS`) are rejected to guard against clock drift. To enforce freshness,
16/// callers can pass `max_age` — or use the `IssuedWithin` policy expression at a higher layer.
17///
18/// **Limitation:** `timestamp` is self-reported by the issuer and can be backdated. For
19/// stronger guarantees, combine with Git commit timestamps or witness receipts.
20///
21/// Args:
22/// * `now`: Current wall-clock time (injected for testability).
23/// * `resolver`: Resolves issuer DIDs to public keys.
24/// * `att`: The attestation to verify.
25/// * `max_age`: If `Some`, rejects attestations older than this duration.
26///
27/// Usage:
28/// ```ignore
29/// // Accept any age:
30/// verify_with_resolver(now, &resolver, &att, None)?;
31/// // Require issued within last hour:
32/// verify_with_resolver(now, &resolver, &att, Some(Duration::hours(1)))?;
33/// ```
34pub fn verify_with_resolver(
35    now: DateTime<Utc>,
36    resolver: &dyn DidResolver,
37    att: &Attestation,
38    max_age: Option<Duration>,
39) -> Result<(), AttestationError> {
40    if att.is_revoked() {
41        return Err(AttestationError::AttestationRevoked);
42    }
43    if let Some(exp) = att.expires_at
44        && now > exp
45    {
46        return Err(AttestationError::AttestationExpired {
47            at: exp.to_rfc3339(),
48        });
49    }
50    if let Some(ts) = att.timestamp
51        && ts > now + Duration::seconds(MAX_SKEW_SECS)
52    {
53        return Err(AttestationError::TimestampInFuture {
54            at: ts.to_rfc3339(),
55        });
56    }
57    if let Some(max) = max_age
58        && let Some(ts) = att.timestamp
59    {
60        let age = now - ts;
61        if age > max {
62            return Err(AttestationError::AttestationTooOld {
63                age_secs: age.num_seconds().unsigned_abs(),
64                max_secs: max.num_seconds().unsigned_abs(),
65            });
66        }
67    }
68
69    // 2. Resolve issuer's public key
70    let resolved = resolver.resolve(&att.issuer).map_err(|e| {
71        AttestationError::DidResolutionError(format!("Resolver error for {}: {}", att.issuer, e))
72    })?;
73    let issuer_pk_bytes = *resolved.public_key();
74
75    // 3. Reconstruct canonical data (MUST match create_with_signatures, includes org fields)
76    let data_to_canonicalize = CanonicalAttestationData {
77        version: att.version,
78        rid: &att.rid,
79        issuer: &att.issuer,
80        subject: &att.subject,
81        device_public_key: att.device_public_key.as_bytes(),
82        payload: &att.payload,
83        timestamp: &att.timestamp,
84        expires_at: &att.expires_at,
85        revoked_at: &att.revoked_at,
86        note: &att.note,
87        role: att.role.as_ref().map(|r| r.as_str()),
88        capabilities: if att.capabilities.is_empty() {
89            None
90        } else {
91            Some(&att.capabilities)
92        },
93        delegated_by: att.delegated_by.as_ref(),
94        signer_type: att.signer_type.as_ref(),
95    };
96    let canonical_json_string = json_canon::to_string(&data_to_canonicalize).map_err(|e| {
97        AttestationError::SerializationError(format!(
98            "Failed to create canonical JSON for verification: {}",
99            e
100        ))
101    })?;
102    let data_to_verify = canonical_json_string.as_bytes();
103    debug!(
104        "(Verify) Canonical data for verification: {}",
105        canonical_json_string
106    );
107
108    // 4. Verify issuer signature
109    let issuer_public_key_ring = UnparsedPublicKey::new(&ED25519, &issuer_pk_bytes);
110    issuer_public_key_ring
111        .verify(data_to_verify, att.identity_signature.as_bytes())
112        .map_err(|e| AttestationError::IssuerSignatureFailed(e.to_string()))?;
113    debug!(
114        "(Verify) Issuer signature verified successfully for {}",
115        att.issuer
116    );
117
118    // 5. Verify subject (device) signature using stored public key
119    let device_public_key_ring = UnparsedPublicKey::new(&ED25519, att.device_public_key.as_bytes());
120    device_public_key_ring
121        .verify(data_to_verify, att.device_signature.as_bytes())
122        .map_err(|e| AttestationError::DeviceSignatureFailed(e.to_string()))?;
123    debug!(
124        "(Verify) Device signature verified successfully for {}",
125        att.subject.as_str()
126    );
127
128    Ok(())
129}

133mod tests {
134    use super::*;
135    use auths_core::signing::{DidResolverError, ResolvedDid};
136    use auths_verifier::IdentityDID;
137    use auths_verifier::core::{Ed25519PublicKey, Ed25519Signature, ResourceId};
138    use auths_verifier::types::DeviceDID;
139
140    struct StubResolver;
141    impl DidResolver for StubResolver {
142        fn resolve(&self, did: &str) -> Result<ResolvedDid, DidResolverError> {
143            Err(DidResolverError::InvalidDidKey(format!("stub: {}", did)))
144        }
145    }
146
147    fn base_attestation() -> Attestation {
148        Attestation {
149            version: 1,
150            rid: ResourceId::new("test"),
151            issuer: IdentityDID::new("did:key:zStub"),
152            subject: DeviceDID::new("did:key:zDevice"),
153            device_public_key: Ed25519PublicKey::from_bytes([0u8; 32]),
154            identity_signature: Ed25519Signature::empty(),
155            device_signature: Ed25519Signature::empty(),
156            revoked_at: None,
157            expires_at: None,
158            timestamp: None,
159            note: None,
160            payload: None,
161            role: None,
162            capabilities: vec![],
163            delegated_by: None,
164            signer_type: None,
165            environment_claim: None,
166        }
167    }
168
169    #[test]
170    fn max_age_none_skips_check() {
171        let mut att = base_attestation();
172        att.timestamp = Some(Utc::now() - Duration::days(365));
173        let result = verify_with_resolver(Utc::now(), &StubResolver, &att, None);
174        // Should pass the timestamp checks and fail on DID resolution (not max_age)
175        assert!(matches!(
176            result,
177            Err(AttestationError::DidResolutionError(_))
178        ));
179    }
180
181    #[test]
182    fn max_age_rejects_old_attestation() {
183        let now = Utc::now();
184        let mut att = base_attestation();
185        att.timestamp = Some(now - Duration::hours(2));
186        let result = verify_with_resolver(now, &StubResolver, &att, Some(Duration::hours(1)));
187        match result {
188            Err(AttestationError::AttestationTooOld { age_secs, max_secs }) => {
189                assert!(age_secs >= 7200);
190                assert_eq!(max_secs, 3600);
191            }
192            other => panic!("Expected AttestationTooOld, got {:?}", other),
193        }
194    }
195
196    #[test]
197    fn max_age_accepts_fresh_attestation() {
198        let now = Utc::now();
199        let mut att = base_attestation();
200        att.timestamp = Some(now - Duration::minutes(30));
201        let result = verify_with_resolver(now, &StubResolver, &att, Some(Duration::hours(1)));
202        // Should pass max_age and fail on DID resolution
203        assert!(matches!(
204            result,
205            Err(AttestationError::DidResolutionError(_))
206        ));
207    }
208
209    #[test]
210    fn max_age_skips_when_no_timestamp() {
211        let att = base_attestation(); // timestamp is None
212        let result =
213            verify_with_resolver(Utc::now(), &StubResolver, &att, Some(Duration::hours(1)));
214        // No timestamp → max_age check skipped, fails on DID resolution
