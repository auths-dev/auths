1//! Headless agent identity provisioning API.
2//!
3//! Provides a library-level API for creating AI agent identities without
4//! interactive prompts. Designed for CI/CD pipelines, orchestration systems,
5//! and daemon processes.
6//!
7//! # Storage Modes
8//!
9//! - [`AgentStorageMode::Persistent`]: Disk-based storage (default: `~/.auths-agent`).
10//!   Agent identity survives process restarts.
11//! - [`AgentStorageMode::InMemory`]: Ephemeral storage for stateless containers
12//!   (Fargate, Docker). Agent identity lives only for the process lifetime.
13//!   Explicitly trades persistence for statelessness.
14//!
15//! # Usage
16//!
17//! ```rust,ignore
18//! use auths_id::agent_identity::{provision_agent_identity, AgentProvisioningConfig, AgentStorageMode};
19//!
20//! let config = AgentProvisioningConfig {
21//!     agent_name: "ci-bot".to_string(),
22//!     capabilities: vec!["sign_commit".to_string()],
23//!     expires_in_secs: Some(86400),
24//!     delegated_by: Some(IdentityDID::new("did:keri:Eabc123")),
25//!     storage_mode: AgentStorageMode::Persistent { repo_path: None },
26//! };
27//!
28//! let keychain = auths_core::storage::keychain::get_platform_keychain()?;
29//! let bundle = provision_agent_identity(config, &my_passphrase_provider, keychain)?;
30//! println!("Agent DID: {}", bundle.agent_did);
31//! ```

55/// Storage mode for agent identity.
56#[derive(Debug, Clone)]
57pub enum AgentStorageMode {
58    /// Persistent storage at a filesystem path.
59    /// Defaults to `~/.auths-agent` if `repo_path` is `None`.
60    Persistent { repo_path: Option<PathBuf> },
61    /// In-memory storage for ephemeral/stateless containers (Fargate, Docker).
62    /// Agent identity lives only for the process lifetime.
63    InMemory,
64}
65
66/// Configuration for provisioning an agent identity.
67#[derive(Debug, Clone)]
68pub struct AgentProvisioningConfig {
69    /// Human-readable agent name (e.g., "ci-bot", "release-agent").
70    pub agent_name: String,
71    /// Capabilities to grant (e.g., `["sign_commit", "pr:create"]`).
72    pub capabilities: Vec<String>,
73    /// Optional expiry in seconds from now.
74    pub expires_in_secs: Option<u64>,
75    /// DID of the human who authorized this agent.
76    pub delegated_by: Option<IdentityDID>,
77    /// Storage mode (persistent or ephemeral).
78    pub storage_mode: AgentStorageMode,
79}

81/// Result of a successful agent provisioning.
82#[derive(Debug, Clone)]
83pub struct AgentIdentityBundle {
84    /// The agent's `did:keri:E...` identity.
85    pub agent_did: IdentityDID,
86    /// The key alias used for signing.
87    pub key_alias: KeyAlias,
88    /// The agent's attestation (with `signer_type: Agent`).
89    pub attestation: Attestation,
90    /// Path to the agent repo (`None` for `InMemory` mode).
91    pub repo_path: Option<PathBuf>,
92}

94/// Errors that can occur during agent provisioning.
95#[derive(Debug, thiserror::Error)]
96#[non_exhaustive]
97pub enum AgentProvisioningError {
98    #[error("repository creation failed: {0}")]
99    RepoCreation(#[from] git2::Error),
100    #[error("identity creation failed: {0}")]
101    IdentityCreation(#[from] crate::error::InitError),
102    #[error("attestation creation failed: {0}")]
103    AttestationCreation(#[from] AttestationError),
104    #[error("keychain access failed: {0}")]
105    KeychainAccess(String),
106    #[error("config write failed: {0}")]
107    ConfigWrite(#[from] std::io::Error),
108}

112/// Provision a new agent identity.
113///
114/// Creates a KERI identity, signs an attestation with `signer_type: Agent`,
115/// and optionally writes an `auths-agent.toml` config file.
116///
117/// Args:
118/// * `backend` - The registry backend for KEL storage. Must be pre-initialized.
119/// * `config` - Provisioning configuration (name, capabilities, storage mode).
120/// * `passphrase_provider` - Plugin point for passphrase retrieval.
121/// * `keychain` - Key storage backend.
122///
123/// Usage:
124/// ```ignore
125/// let bundle = provision_agent_identity(Arc::new(my_backend), config, &provider, keychain)?;
126/// ```
127pub fn provision_agent_identity(
128    now: DateTime<Utc>,
129    backend: Arc<dyn RegistryBackend + Send + Sync>,
130    config: AgentProvisioningConfig,
131    passphrase_provider: &dyn PassphraseProvider,
132    keychain: Box<dyn KeyStorage + Send + Sync>,
133) -> Result<AgentIdentityBundle, AgentProvisioningError> {
134    let (repo_path, ephemeral) = resolve_repo_path(&config.storage_mode)?;
135    ensure_git_repo(&repo_path)?;
136
137    let key_alias = key_alias_for(&config.storage_mode);
138    let agent_did = get_or_create_identity(
139        backend,
140        &key_alias,
141        &config,
142        passphrase_provider,
143        &*keychain,
144    )?;
145
146    let attestation = sign_agent_attestation(
147        now,
148        &agent_did,
149        &key_alias,
150        &config,
151        passphrase_provider,
152        keychain,
153    )?;
154
155    if !ephemeral {
156        write_agent_toml(&repo_path, agent_did.as_str(), key_alias.as_str(), &config)?;
157    }
158
159    Ok(AgentIdentityBundle {
160        agent_did,
161        key_alias,
162        attestation,
163        repo_path: if ephemeral { None } else { Some(repo_path) },
164    })
165}

169/// Resolve the repo path from storage mode. Returns `(path, is_ephemeral)`.
170fn resolve_repo_path(mode: &AgentStorageMode) -> Result<(PathBuf, bool), AgentProvisioningError> {
171    match mode {
172        AgentStorageMode::Persistent { repo_path } => {
173            let path = match repo_path {
174                Some(p) => p.clone(),
175                None => default_agent_repo_path()?,
176            };
177            Ok((path, false))
178        }
179        AgentStorageMode::InMemory => {
180            // Leak the tempdir so it persists for the process lifetime.
181            let tmp = tempfile::tempdir().map_err(AgentProvisioningError::ConfigWrite)?;
182            let path = tmp.path().to_path_buf();
183            // Leak the tempdir so cleanup doesn't run — ephemeral agents persist for process lifetime.
184            std::mem::forget(tmp);
185            Ok((path, true))
186        }
187    }
188}
189
190fn ensure_git_repo(path: &Path) -> Result<(), AgentProvisioningError> {
191    if !path.exists() {
192        std::fs::create_dir_all(path)?;
193    }
194    if git2::Repository::open(path).is_err() {
195        git2::Repository::init(path)?;
196    }
197    Ok(())
198}
199
200fn default_agent_repo_path() -> Result<PathBuf, AgentProvisioningError> {
201    let home = dirs::home_dir().ok_or_else(|| {
202        AgentProvisioningError::ConfigWrite(std::io::Error::new(
203            std::io::ErrorKind::NotFound,
204            "could not determine home directory",
205        ))
206    })?;
207    Ok(home.join(".auths-agent"))
208}
209
210fn key_alias_for(mode: &AgentStorageMode) -> KeyAlias {
211    match mode {
212        AgentStorageMode::Persistent { .. } => KeyAlias::new_unchecked("agent-key"),
213        AgentStorageMode::InMemory => KeyAlias::new_unchecked("agent-key-ephemeral"),
214    }
215}

219/// Return the existing identity DID or create a new one.
220fn get_or_create_identity(
221    backend: Arc<dyn RegistryBackend + Send + Sync>,
222    key_alias: &KeyAlias,
223    _config: &AgentProvisioningConfig,
224    passphrase_provider: &dyn PassphraseProvider,
225    keychain: &(dyn KeyStorage + Send + Sync),
226) -> Result<IdentityDID, AgentProvisioningError> {
227    let mut existing_did: Option<IdentityDID> = None;
228    let _ = backend.visit_identities(&mut |prefix| {
229        existing_did = Some(IdentityDID::new(format!("did:keri:{}", prefix)));
230        std::ops::ControlFlow::Break(())
231    });
232    if let Some(did) = existing_did {
233        return Ok(did);
234    }
235
236    let (did, _) =
237        initialize_registry_identity(backend, key_alias, passphrase_provider, keychain, None)?;
238
239    Ok(did)
240}

244/// Create and sign an attestation with `signer_type: Agent`.
245///
246/// The flow:
247/// 1. Decrypt the key to extract the device public key
248/// 2. Create a base attestation via `create_signed_attestation`
249/// 3. Stamp `signer_type` and `delegated_by`
250/// 4. Re-sign so the canonical data covers the new fields
251fn sign_agent_attestation(
252    now: DateTime<Utc>,
253    controller_did: &IdentityDID,
254    key_alias: &KeyAlias,
255    config: &AgentProvisioningConfig,
256    passphrase_provider: &dyn PassphraseProvider,
257    keychain: Box<dyn KeyStorage + Send + Sync>,
258) -> Result<Attestation, AgentProvisioningError> {
259    let device_pk = extract_public_key(key_alias, passphrase_provider, &*keychain)?;
260    let device_did = DeviceDID::from_ed25519(&device_pk);
261    let meta = build_attestation_meta(now, config);
262    let signer = StorageSigner::new(keychain);
263
264    let rid = format!("agent:{}", config.agent_name);
265    let mut att = create_signed_attestation(
266        now,
267        &rid,
268        controller_did,
269        &device_did,
270        &device_pk,
271        None,
272        &meta,
273        &signer,
274        passphrase_provider,
275        Some(key_alias),
276        Some(key_alias),
277        vec![],
278        None,
279        config.delegated_by.clone(),
280    )?;
281
282    att.signer_type = Some(SignerType::Agent);
283    resign_attestation(
284        &mut att,
285        &signer,
286        passphrase_provider,
287        Some(key_alias),
288        key_alias,
289    )?;
290
291    Ok(att)
292}

294/// Decrypt the stored key and return the 32-byte Ed25519 public key.
295fn extract_public_key(
296    key_alias: &KeyAlias,
297    passphrase_provider: &dyn PassphraseProvider,
298    keychain: &dyn KeyStorage,
299) -> Result<[u8; 32], AgentProvisioningError> {
300    let (_did, _role, encrypted) = keychain
301        .load_key(key_alias)
302        .map_err(|e| AgentProvisioningError::KeychainAccess(e.to_string()))?;
303
304    let passphrase = passphrase_provider
305        .get_passphrase("agent key passphrase")
306        .map_err(|e| AgentProvisioningError::KeychainAccess(e.to_string()))?;
307
308    let decrypted = decrypt_keypair(&encrypted, &passphrase)
309        .map_err(|e| AgentProvisioningError::KeychainAccess(e.to_string()))?;
310
311    let kp = crate::identity::helpers::load_keypair_from_der_or_seed(&decrypted)
312        .map_err(|e| AgentProvisioningError::KeychainAccess(format!("bad pkcs8: {}", e)))?;
313
314    let pk: [u8; 32] = kp
315        .public_key()
316        .as_ref()
317        .try_into()
318        .map_err(|_| AgentProvisioningError::KeychainAccess("unexpected key length".into()))?;
319
320    Ok(pk)
321}

323fn build_attestation_meta(
324    now: DateTime<Utc>,
325    config: &AgentProvisioningConfig,
326) -> AttestationMetadata {
327    let expires_at = config
328        .expires_in_secs
329        .map(|s| now + chrono::Duration::seconds(s as i64));
330
331    AttestationMetadata {
332        note: Some(format!("Agent: {}", config.agent_name)),
333        timestamp: Some(now),
334        expires_at,
335    }
336}

340fn write_agent_toml(
341    repo_path: &Path,
342    did: &str,
343    key_alias: &str,
344    config: &AgentProvisioningConfig,
345) -> Result<(), AgentProvisioningError> {
346    let content = format_agent_toml(did, key_alias, config);
347    std::fs::write(repo_path.join("auths-agent.toml"), content)?;
348    Ok(())
349}
350
351pub fn format_agent_toml(did: &str, key_alias: &str, config: &AgentProvisioningConfig) -> String {
352    let caps = config
353        .capabilities
354        .iter()
355        .map(|c| format!("\"{}\"", c))
356        .collect::<Vec<_>>()
357        .join(", ");
358
359    let mut out = format!(
360        "# Auths Agent Configuration\n\
361         # Generated by provision_agent_identity()\n\n\
362         [agent]\n\
363         name = \"{}\"\n\
364         did = \"{}\"\n\
365         key_alias = \"{}\"\n\
366         signer_type = \"Agent\"\n",
367        config.agent_name, did, key_alias,
368    );
369
370    if let Some(ref delegator) = config.delegated_by {
371        out.push_str(&format!("delegated_by = \"{}\"\n", delegator));
372    }
373
374    out.push_str(&format!("\n[capabilities]\ngranted = [{}]\n", caps));
375
376    if let Some(secs) = config.expires_in_secs {
377        out.push_str(&format!("\n[expiry]\nexpires_in_secs = {}\n", secs));
378    }
379
380    out
381}

389    #[test]
390    fn format_agent_toml_with_all_fields() {
391        let config = AgentProvisioningConfig {
392            agent_name: "ci-bot".to_string(),
393            capabilities: vec!["sign_commit".to_string(), "pr:create".to_string()],
394            expires_in_secs: Some(86400),
395            delegated_by: Some(IdentityDID::new("did:keri:Eabc123")),
396            storage_mode: AgentStorageMode::Persistent { repo_path: None },
397        };
398        let toml = format_agent_toml("did:keri:Eagent", "agent-key", &config);
399        assert!(toml.contains("name = \"ci-bot\""));
400        assert!(toml.contains("did = \"did:keri:Eagent\""));
401        assert!(toml.contains("delegated_by = \"did:keri:Eabc123\""));
402        assert!(toml.contains("\"sign_commit\", \"pr:create\""));
403        assert!(toml.contains("expires_in_secs = 86400"));
404    }
