1//! Provides storage implementation for KERI Key Event Logs (KELs) within a Git repository.
2//! e.g., `refs/did/keri/<prefix>/kel`
3
4use crate::error::StorageError;
5use crate::identity::events::KeyRotationEvent;
6use crate::keri::Prefix;
7use crate::storage::layout;
8use git2::{Commit, ErrorCode, Oid, Repository, Signature};
9use log::{debug, warn};
10use sha2::{Digest, Sha256};
11use std::path::PathBuf;

13/// Standard filename for storing KERI event data within commit blobs in the KEL.
14const KERI_EVENT_BLOB_NAME: &str = "event.cesr"; // Or "event.json", "event.cbor" etc.
15
16/// Provides methods for interacting with KERI Key Event Logs stored in a Git repository.
17#[derive(Debug, Clone)]
18pub struct KeriGitStorage {
19    /// Path to the root of the Git repository.
20    repo_path: PathBuf,
21}
22
23impl KeriGitStorage {
24    /// Creates a new `KeriGitStorage` instance for the given repository path.
25    pub fn new(repo_path: impl Into<PathBuf>) -> Self {
26        KeriGitStorage {
27            repo_path: repo_path.into(),
28        }
29    }
30
31    fn open_repo(&self) -> Result<Repository, StorageError> {
32        Ok(Repository::open(&self.repo_path)?)
33    }

35    /// Stores a single KERI event as a new commit on the specific KEL Git reference
36    /// for the given DID prefix (AID).
37    ///
38    /// # Arguments
39    /// * `did_prefix`: The KERI Autonomous Identifier (AID) prefix.
40    /// * `event_bytes`: The serialized KERI event data (e.g., CESR).
41    /// * `commit_message`: A descriptive message for the Git commit (e.g., "Inception", "Rotation").
42    ///
43    /// # Returns
44    /// The `Oid` of the newly created Git commit storing the event.
45    pub fn store_event(
46        &self,
47        did_prefix: &Prefix,
48        event_bytes: &[u8],
49        commit_message: &str,
50        now: chrono::DateTime<chrono::Utc>,
51    ) -> Result<Oid, StorageError> {
52        debug!(
53            "Storing KERI event for prefix '{}' in repo {:?}. Message: '{}'",
54            did_prefix.as_str(),
55            self.repo_path,
56            commit_message
57        );
58        let repo = self.open_repo()?;
59
60        // 1. Determine the specific KEL reference path using the layout helper
61        let kel_ref_name = layout::keri_kel_ref(did_prefix);
62        debug!("Target KEL ref: {}", kel_ref_name);
63
64        // 2. Find the parent commit (the current tip of this specific KEL ref)
65        let parent_commit_opt = match repo.find_reference(&kel_ref_name) {
66            Ok(reference) => {
67                // Ref exists, peel it to a commit
68                match reference.peel_to_commit() {
69                    Ok(commit) => {
70                        debug!(
71                            "Found parent commit {} for ref '{}'",
72                            commit.id(),
73                            kel_ref_name
74                        );
75                        Some(commit)
76                    }
77                    Err(e)
78                        if e.code() == ErrorCode::Peel
79                            || e.code() == ErrorCode::NotFound
80                            || e.code() == ErrorCode::InvalidSpec =>
81                    {
82                        // Ref exists but doesn't point to a valid commit (e.g., symbolic, broken, unborn)
83                        warn!(
84                            "Ref '{}' exists but doesn't point to a valid commit ({:?}). Creating commit without parent.",
85                            kel_ref_name,
86                            e.code()
87                        );
88                        None
89                    }
90                    Err(e) => return Err(e.into()),
91                }
92            }
93            Err(e) if e.code() == ErrorCode::NotFound => {
94                // Ref doesn't exist, this is the initial commit for this KEL (no parent)
95                debug!(
96                    "KEL reference '{}' not found. Creating initial commit.",
97                    kel_ref_name
98                );
99                None
100            }
101            Err(e) => return Err(e.into()),
102        };
103        let parents: Vec<&Commit> = parent_commit_opt.iter().collect();
104
105        // 3. Create the Git blob containing the event data
106        let blob_oid = repo.blob(event_bytes)?;
107        debug!("Created event blob: {}", blob_oid);
108
109        // 4. Create the Git tree containing the blob
110        let mut tree_builder = repo.treebuilder(None)?;
111        tree_builder.insert(KERI_EVENT_BLOB_NAME, blob_oid, 0o100644)?;
112        let tree_oid = tree_builder.write()?;
113        let tree = repo.find_tree(tree_oid)?;
114        debug!("Created tree {} containing event blob", tree_oid);
115
116        // 5. Create the Git signature (use repo default or fallback)
117        let sig = repo.signature().or_else(|_| {
118            Signature::new(
119                "auths-keri",
120                "auths-keri@localhost",
121                &git2::Time::new(now.timestamp(), 0),
122            )
123        })?;
124        debug!("Using Git signature: {}", sig);
125
126        // 6. Create the Git commit object
127        let commit_oid = repo.commit(None, &sig, &sig, commit_message, &tree, &parents)?;
128        debug!("Created commit object: {}", commit_oid);
129
130        // 7. Explicitly update the specific KEL Git reference to point to the new commit
131        //    Use force=true to overwrite if the reference already exists (it should if parent was found).
132        let ref_log_message = format!("commit (keri event): {}", commit_message);
133        repo.reference(&kel_ref_name, commit_oid, true, &ref_log_message)?;
134        debug!(
135            "Updated KEL reference '{}' to point to {}",
136            kel_ref_name, commit_oid
137        );
138
139        Ok(commit_oid)
140    }

142    /// Reads the history of KERI events for a specific DID prefix (AID) from its KEL Git reference.
143    /// Returns the events as raw bytes, ordered from oldest (inception) to newest.
144    ///
145    /// # Arguments
146    /// * `did_prefix`: The KERI Autonomous Identifier (AID) prefix.
147    ///
148    /// # Returns
149    /// A `Vec` containing the raw bytes (`Vec<u8>`) of each KERI event found in the KEL history,
150    /// ordered chronologically (oldest first). Returns an empty `Vec` if the KEL ref is not found.
151    pub fn read_kel_history(&self, did_prefix: &Prefix) -> Result<Vec<Vec<u8>>, StorageError> {
152        debug!(
153            "Reading KEL history for prefix '{}' from repo {:?}",
154            did_prefix.as_str(),
155            self.repo_path
156        );
157        let repo = self.open_repo()?;
158
159        // 1. Determine the specific KEL reference path
160        let kel_ref_name = layout::keri_kel_ref(did_prefix);
161        debug!("Target KEL ref: {}", kel_ref_name);
162
163        // 2. Find the starting commit (tip of the KEL ref)
164        let start_commit = match repo.find_reference(&kel_ref_name) {
165            Ok(reference) => {
166                // Ref exists, peel to commit
167                reference.peel_to_commit()?
168            }
169            Err(e) if e.code() == ErrorCode::NotFound => {
170                // Ref doesn't exist, no history for this prefix
171                debug!(
172                    "KEL reference '{}' not found. Returning empty history.",
173                    kel_ref_name
174                );
175                return Ok(Vec::new()); // Return empty vector, not an error
176            }
177            Err(e) => return Err(e.into()),
178        };
179        debug!("Starting history walk from commit {}", start_commit.id());
180
181        // 3. Walk the Git history backwards (first parent) and collect event blobs
182        let mut event_history_bytes = Vec::new();
183        let mut current_commit = start_commit;
184
185        loop {
186            // Get the tree for the current commit
187            match current_commit.tree() {
188                Ok(tree) => {
189                    // Try to find the event blob within the tree
190                    match tree.get_name(KERI_EVENT_BLOB_NAME) {
191                        Some(entry) => {
192                            // Blob entry found, find the blob object
193                            match repo.find_blob(entry.id()) {
194                                Ok(blob) => {
195                                    // Successfully found blob, add its content to history
196                                    event_history_bytes.push(blob.content().to_vec());
197                                    debug!(
198                                        "Read event blob {} from commit {}",
199                                        blob.id(),
200                                        current_commit.id()
201                                    );
202                                }
203                                Err(e) => {
204                                    // Blob OID found in tree, but blob object not found (repo corruption?)
205                                    warn!(
206                                        "Failed to find blob object {} referenced in commit {}: {}. Skipping event.",
207                                        entry.id(),
208                                        current_commit.id(),
209                                        e
210                                    );
211                                    // Decide whether to error out or just skip. Skipping makes it more robust.
212                                }
213                            }
214                        }
215                        None => {
216                            // Commit exists, but doesn't contain the expected event blob
217                            warn!(
218                                "Commit {} in KEL history for '{}' does not contain the expected blob '{}'. Skipping commit.",
219                                current_commit.id(),
220                                did_prefix.as_str(),
221                                KERI_EVENT_BLOB_NAME
222                            );
223                            // Skip this commit and continue walking
224                        }
225                    }
226                }
227                Err(e) => {
228                    // Failed to get tree for the commit (repo issue?)
229                    warn!(
230                        "Failed to get tree for commit {} in KEL history for '{}': {}. Stopping history walk early.",
231                        current_commit.id(),
232                        did_prefix.as_str(),
233                        e
234                    );
235                    // Stop walking if tree is inaccessible, history might be incomplete
236                    break;
237                }
238            }
239
240            // Move to the first parent commit
241            match current_commit.parent_count() {
242                0 => break, // Reached the root of this line of history
243                _ => {
244                    match current_commit.parent(0) {
245                        Ok(parent) => current_commit = parent,
246                        Err(e) => {
247                            // Failed to get parent commit (repo issue?)
248                            warn!(
249                                "Failed to get parent commit for {} in KEL history for '{}': {}. Stopping history walk early.",
250                                current_commit.id(),
251                                did_prefix.as_str(),
252                                e
253                            );
254                            // Stop walking if parent is inaccessible, history might be incomplete
255                            break;
256                        }
257                    }
258                }
259            }
260        } // End loop
261
262        // 4. Reverse the collected bytes so the oldest event (inception) is first
263        event_history_bytes.reverse();
264        debug!(
265            "Finished reading KEL history for '{}'. Found {} events.",
266            did_prefix.as_str(),
267            event_history_bytes.len()
268        );
269
270        Ok(event_history_bytes)
271    }

273    /// Appends a key rotation event to the KEL, validating chain integrity.
274    ///
275    /// This method verifies that the `previous_hash` in the event matches the
276    /// hash of the current KEL tip commit, ensuring the rotation event is
277    /// properly chained.
278    ///
279    /// # Arguments
280    /// * `did_prefix` - The KERI Autonomous Identifier (AID) prefix.
281    /// * `event` - The KeyRotationEvent to append.
282    ///
283    /// # Returns
284    /// * `Ok(Oid)` - The Git commit OID of the stored rotation event.
285    /// * `Err(...)` - If validation fails or storage fails.
286    pub fn append_rotation_event(
287        &self,
288        did_prefix: &Prefix,
289        event: &KeyRotationEvent,
290        now: chrono::DateTime<chrono::Utc>,
291    ) -> Result<Oid, StorageError> {
292        debug!(
293            "Appending rotation event (seq {}) for prefix '{}' in repo {:?}",
294            event.sequence,
295            did_prefix.as_str(),
296            self.repo_path
297        );
298
299        let repo = self.open_repo()?;
300        let kel_ref_name = layout::keri_kel_ref(did_prefix);
301
302        // 1. Get the current KEL tip commit hash (if exists)
303        let current_tip_hash = match repo.find_reference(&kel_ref_name) {
304            Ok(reference) => {
305                match reference.peel_to_commit() {
306                    Ok(commit) => {
307                        // Compute hash of the commit ID for chain verification
308                        let commit_id = commit.id().to_string();
309                        let mut hasher = Sha256::new();
310                        hasher.update(commit_id.as_bytes());
311                        let hash = format!("{:x}", hasher.finalize());
312                        debug!("Current KEL tip commit: {}, hash: {}", commit.id(), hash);
313                        Some(hash)
314                    }
315                    Err(_) => None,
316                }
317            }
318            Err(e) if e.code() == ErrorCode::NotFound => {
319                debug!(
320                    "No existing KEL for prefix '{}', this should be the first event",
321                    did_prefix.as_str()
322                );
323                None
324            }
325            Err(e) => return Err(e.into()),
326        };
327
328        // 2. Validate chain integrity
329        if event.sequence == 0 {
330            if !event.previous_hash.is_empty() && current_tip_hash.is_some() {
331                return Err(StorageError::InvalidData(format!(
332                    "Sequence 0 event (inception) should have empty previous_hash, got: {}",
333                    event.previous_hash
334                )));
335            }
336        } else {
337            match &current_tip_hash {
338                Some(expected) if &event.previous_hash != expected => {
339                    return Err(StorageError::InvalidData(format!(
340                        "Chain integrity violation: event.previous_hash ({}) != current KEL tip hash ({})",
341                        event.previous_hash, expected
342                    )));
343                }
344                None => {
345                    return Err(StorageError::InvalidData(format!(
346                        "Cannot append rotation event (seq {}) without existing KEL history",
347                        event.sequence
348                    )));
349                }
350                _ => {
351                    debug!("Chain integrity validated for sequence {}", event.sequence);
352                }
353            }
354        }
355
356        // 3. Serialize the event as JSON
357        let event_json = serde_json::to_vec_pretty(event)?;
358
359        // 4. Store the event using the existing store_event method
360        let commit_message = format!("Key rotation: sequence {}", event.sequence);
361        self.store_event(did_prefix, &event_json, &commit_message, now)
362    }
363}
