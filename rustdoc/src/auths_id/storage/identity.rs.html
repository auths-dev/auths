4/// Trait for abstracting the storage and retrieval of identity information.
5///
6/// Implementations handle the underlying storage mechanism (e.g., Git repository)
7/// and use a `StorageLayoutConfig` to determine specific paths and filenames.
8pub trait IdentityStorage {
9    /// Creates or updates the identity reference (defined in config) with the
10    /// controller DID and optional, arbitrary metadata.
11    ///
12    /// The structure and interpretation of the `metadata` JSON is the responsibility
13    /// of the caller. This function stores the provided `controller_did` and `metadata`
14    /// generically in a blob (name defined in config).
15    ///
16    /// # Arguments
17    /// * `controller_did`: The DID string controlling this identity.
18    /// * `metadata`: Optional arbitrary JSON value representing identity metadata.
19    fn create_identity(
20        &self,
21        controller_did: &str,
22        metadata: Option<serde_json::Value>,
23    ) -> Result<(), StorageError>;
24
25    /// Loads the identity information (controller DID, metadata, storage ID)
26    /// from the configured identity reference and blob name.
27    ///
28    /// Returns a `ManagedIdentity` struct containing the loaded `controller_did`,
29    /// the storage identifier (e.g., repository name), and the full `metadata`
30    /// field as a `serde_json::Value` for the caller to interpret.
31    fn load_identity(&self) -> Result<ManagedIdentity, StorageError>;
32
33    /// Gets the configured primary Git reference used for storing the identity commit.
34    fn get_identity_ref(&self) -> Result<String, StorageError>;
35}
