1//! Org membership validation and listing API.
2//!
3//! This module provides types and utilities for validating and filtering org members.
4//! It replaces filename-based enumeration with validation-aware APIs that:
5//! - Parse and validate JSON attestations, not just filenames
6//! - Compute revoked/expired/invalid status in one pass
7//! - Provide clear contracts for why entries are invalid
8//!
9//! ## Domain Types (Sans-IO)
10//!
11//! All types in this module are **pure domain types** with no I/O:
12//! - `MemberFilter`: Time is injected via `now: Option<DateTime<Utc>>` (never uses `Utc::now()`)
13//! - `MemberStatus`, `MemberStatusKind`: Pure enums
14//! - `MemberView`, `OrgMemberEntry`: Pure data structures
15//! - `compute_status()`: Pure function taking `now` as parameter
16//!
17//! ## Structural Invariants (checked during visit)
18//!
19//! These are **hard errors** that make an entry Invalid:
20//! - JSON must parse
21//! - `attestation.subject` must match filename DID
22//! - `attestation.issuer` must match expected org issuer (`did:keri:{org}`)
23//!
24//! ## Runtime Status (computed from valid attestations)
25//!
26//! - Active: not revoked, not expired
27//! - Revoked: `att.is_revoked()` (i.e. `revoked_at` is `Some`)
28//! - Expired: `att.expires_at <= now`
29

37/// Filter for querying org members.
38#[derive(Debug, Clone)]
39pub struct MemberFilter {
40    /// Which statuses to include (default: {Active})
41    pub include_statuses: HashSet<MemberStatusKind>,
42    /// Optional role whitelist - include if member.role is in set
43    pub roles_any: Option<HashSet<Role>>,
44    /// Include if intersection(member_caps, filter_caps) non-empty
45    pub capabilities_any: Option<HashSet<Capability>>,
46    /// Include only if filter_caps ⊆ member_caps
47    pub capabilities_all: Option<HashSet<Capability>>,
48    /// Injectable timestamp for deterministic tests
49    pub now: Option<DateTime<Utc>>,
50}
51
52// Manual Default: derive(Default) would give empty HashSet = no members
53impl Default for MemberFilter {
54    fn default() -> Self {
55        let mut include_statuses = HashSet::new();
56        include_statuses.insert(MemberStatusKind::Active);
57        Self {
58            include_statuses,
59            roles_any: None,
60            capabilities_any: None,
61            capabilities_all: None,
62            now: None,
63        }
64    }
65}

67/// Coarse status for filtering.
68#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
69pub enum MemberStatusKind {
70    Active,
71    Revoked,
72    Expired,
73    Invalid,
74}
75
76impl MemberStatusKind {
77    /// Rank for deterministic sorting (lower = first)
78    pub fn rank(self) -> u8 {
79        match self {
80            MemberStatusKind::Active => 0,
81            MemberStatusKind::Revoked => 1,
82            MemberStatusKind::Expired => 2,
83            MemberStatusKind::Invalid => 3,
84        }
85    }
86}

88/// Detailed status for UI/debugging.
89#[derive(Debug, Clone, PartialEq, Eq)]
90pub enum MemberStatus {
91    Active,
92    Revoked,
93    Expired { expires_at: DateTime<Utc> },
94    Invalid { reason: MemberInvalidReason },
95}
96
97impl MemberStatus {
98    pub fn kind(&self) -> MemberStatusKind {
99        match self {
100            MemberStatus::Active => MemberStatusKind::Active,
101            MemberStatus::Revoked => MemberStatusKind::Revoked,
102            MemberStatus::Expired { .. } => MemberStatusKind::Expired,
103            MemberStatus::Invalid { .. } => MemberStatusKind::Invalid,
104        }
105    }
106}

108/// Why a member entry is invalid (structural corruption).
109///
110/// These are **hard errors** representing structural corruption, not soft warnings.
111/// An entry with any of these reasons should be treated as if it doesn't exist
112/// for authorization purposes.
113///
114/// # Protocol Invariants
115///
116/// - **Subject Integrity**: `attestation.subject` must match the filename DID
117/// - **Issuer Integrity**: `attestation.issuer` must match `did:keri:{org}`
118///
119/// Violating these indicates either data corruption or an attack attempt.
120#[derive(Debug, Clone, PartialEq, Eq)]
121pub enum MemberInvalidReason {
122    /// JSON could not be parsed.
123    JsonParseError(String),
124
125    /// The attestation's subject DID doesn't match the filename DID.
126    ///
127    /// This is a **structural integrity violation**. The filename determines
128    /// which device/member this attestation is for, and the attestation's
129    /// subject field must match.
130    SubjectMismatch {
131        filename_did: DeviceDID,
132        attestation_subject: DeviceDID,
133    },
134
135    /// The attestation's issuer DID doesn't match the expected org issuer.
136    ///
137    /// This is a **structural integrity violation**. Org member attestations
138    /// must be issued by `did:keri:{org}`. An attestation with a different
139    /// issuer doesn't prove membership in this org.
140    IssuerMismatch {
141        expected_issuer: IdentityDID,
142        actual_issuer: IdentityDID,
143    },
144
145    /// Other validation errors (e.g., file read errors).
146    Other(String),
147}

157/// High-level view for UI/CLI.
158///
159/// # Status Computation
160///
161/// For valid attestations, `status` is set to `Active` by the backend.
162/// The backend does NOT compute revoked/expired status (that's policy).
163///
164/// To compute actual status, use:
165/// ```ignore
166/// let status = if view.revoked_at.is_some() {
167///     MemberStatus::Revoked
168/// } else if view.expires_at.map(|e| e <= now).unwrap_or(false) {
169///     MemberStatus::Expired { expires_at: view.expires_at.unwrap() }
170/// } else {
171///     MemberStatus::Active
172/// };
173/// ```
174///
175/// Or use `compute_status()` from this module with the raw attestation.
176#[derive(Debug, Clone)]
177pub struct MemberView {
178    pub did: DeviceDID,
179    /// Status: Active for valid attestations, Invalid for structural errors.
180    /// Note: Backend does NOT compute Revoked/Expired - use `revoked_at` and
181    /// `expires_at` fields to compute these statuses in the policy layer.
182    pub status: MemberStatus,
183    pub role: Option<Role>,
184    pub capabilities: Vec<Capability>,
185    pub issuer: IdentityDID,
186    pub rid: ResourceId,
187    /// Timestamp when the attestation was revoked (`None` if not revoked).
188    pub revoked_at: Option<DateTime<Utc>>,
189    pub expires_at: Option<DateTime<Utc>>,
190    pub timestamp: Option<DateTime<Utc>>,
191    /// Original filename for debugging invalid entries
192    pub source_filename: String,
193}
194
195/// Compute status from attestation. Uses <= for expiry boundary.
196pub fn compute_status(att: &Attestation, now: DateTime<Utc>) -> MemberStatus {
197    if att.is_revoked() {
198        MemberStatus::Revoked
199    } else if let Some(expires_at) = att.expires_at {
200        if expires_at <= now {
201            MemberStatus::Expired { expires_at }
202        } else {
203            MemberStatus::Active
204        }
205    } else {
206        MemberStatus::Active
207    }
208}

210/// Convert a Capability to its canonical string representation.
211///
212/// This delegates to `Capability::to_string()` which is the authoritative
213/// string form. Used for both filtering and display.
214pub fn capability_to_string(cap: &auths_verifier::core::Capability) -> String {
215    cap.to_string()
216}
217
218/// Get capability strings from an attestation as a HashSet (for filtering).
219pub fn attestation_capability_strings(att: &Attestation) -> HashSet<String> {
220    att.capabilities.iter().map(capability_to_string).collect()
221}
222
223/// Get capability strings from an attestation as a Vec (for display).
224pub fn attestation_capability_vec(att: &Attestation) -> Vec<String> {
225    att.capabilities.iter().map(capability_to_string).collect()
226}

228/// Compute the expected issuer DID for an org.
229pub fn expected_org_issuer(org: &str) -> String {
230    format!("did:keri:{}", org)
231}
