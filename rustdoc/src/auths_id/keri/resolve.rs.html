1//! did:keri resolution via KEL replay.
2//!
3//! Resolves a `did:keri:<prefix>` to its current public key by:
4//! 1. Loading the KEL from Git
5//! 2. Replaying all events to derive current KeyState
6//! 3. Decoding the current public key
7
8use auths_crypto::KeriPublicKey;
9use git2::Repository;

11use super::types::Prefix;
12use super::{GitKel, KelError, ValidationError, validate_kel};

14/// Error type for did:keri resolution.
15#[derive(Debug, thiserror::Error)]
16#[non_exhaustive]
17pub enum ResolveError {
18    #[error("Invalid DID format: {0}")]
19    InvalidFormat(String),
20
21    #[error("KEL not found for prefix: {0}")]
22    NotFound(String),
23
24    #[error("KEL error: {0}")]
25    Kel(#[from] KelError),
26
27    #[error("Validation error: {0}")]
28    Validation(#[from] ValidationError),
29
30    #[error("Invalid key encoding: {0}")]
31    InvalidKeyEncoding(String),
32
33    #[error("No current key in identity")]
34    NoCurrentKey,
35
36    #[error("Unknown key type: {0}")]
37    UnknownKeyType(String),
38}
39
40/// Result of resolving a did:keri.
41#[derive(Debug, Clone)]
42pub struct DidKeriResolution {
43    /// The full DID string
44    pub did: String,
45
46    /// The KERI prefix
47    pub prefix: Prefix,
48
49    /// The current public key (raw bytes, 32 bytes for Ed25519)
50    pub public_key: Vec<u8>,
51
52    /// The current sequence number
53    pub sequence: u64,
54
55    /// Whether the identity can still be rotated
56    pub can_rotate: bool,
57
58    /// Whether the identity has been abandoned
59    pub is_abandoned: bool,
60}

62/// Resolve a did:keri to its current public key.
63///
64/// This replays the entire KEL to derive the current key state.
65///
66/// # Arguments
67/// * `repo` - Git repository containing the KEL
68/// * `did` - The did:keri string (e.g., "did:keri:EXq5YqaL...")
69///
70/// # Returns
71/// * `DidKeriResolution` with the current public key and state
72pub fn resolve_did_keri(repo: &Repository, did: &str) -> Result<DidKeriResolution, ResolveError> {
73    let prefix = parse_did_keri(did)?;
74
75    // Load KEL
76    let kel = GitKel::new(repo, prefix.as_str());
77    if !kel.exists() {
78        return Err(ResolveError::NotFound(prefix.as_str().to_string()));
79    }
80
81    // Replay KEL to get current state
82    let events = kel.get_events()?;
83    let state = validate_kel(&events)?;
84
85    // Decode current public key
86    let key_encoded = state.current_key().ok_or(ResolveError::NoCurrentKey)?;
87
88    let public_key = KeriPublicKey::parse(key_encoded)
89        .map(|k| k.as_bytes().to_vec())
90        .map_err(|e| ResolveError::InvalidKeyEncoding(e.to_string()))?;
91
92    Ok(DidKeriResolution {
93        did: did.to_string(),
94        prefix,
95        public_key,
96        sequence: state.sequence,
97        can_rotate: state.can_rotate(),
98        is_abandoned: state.is_abandoned,
99    })
100}

102/// Resolve a did:keri at a specific sequence number (historical lookup).
103///
104/// This replays the KEL only up to the target sequence.
105///
106/// # Arguments
107/// * `repo` - Git repository containing the KEL
108/// * `did` - The did:keri string
109/// * `target_sequence` - The sequence number to resolve at
110pub fn resolve_did_keri_at_sequence(
111    repo: &Repository,
112    did: &str,
113    target_sequence: u64,
114) -> Result<DidKeriResolution, ResolveError> {
115    let prefix = parse_did_keri(did)?;
116
117    let kel = GitKel::new(repo, prefix.as_str());
118    if !kel.exists() {
119        return Err(ResolveError::NotFound(prefix.as_str().to_string()));
120    }
121
122    let events = kel.get_events()?;
123
124    // Only process events up to target sequence
125    let events_subset: Vec<_> = events
126        .into_iter()
127        .take_while(|e| e.sequence().value() <= target_sequence)
128        .collect();
129
130    if events_subset.is_empty() {
131        return Err(ResolveError::NotFound(format!(
132            "No events at sequence {}",
133            target_sequence
134        )));
135    }
136
137    let state = validate_kel(&events_subset)?;
138
139    let key_encoded = state.current_key().ok_or(ResolveError::NoCurrentKey)?;
140    let public_key = KeriPublicKey::parse(key_encoded)
141        .map(|k| k.as_bytes().to_vec())
142        .map_err(|e| ResolveError::InvalidKeyEncoding(e.to_string()))?;
143
144    Ok(DidKeriResolution {
145        did: did.to_string(),
146        prefix,
147        public_key,
148        sequence: state.sequence,
149        can_rotate: state.can_rotate(),
150        is_abandoned: state.is_abandoned,
151    })
152}

154/// Parse a did:keri string to extract the prefix.
155pub fn parse_did_keri(did: &str) -> Result<Prefix, ResolveError> {
156    const PREFIX: &str = "did:keri:";
157
158    if !did.starts_with(PREFIX) {
159        return Err(ResolveError::InvalidFormat(format!(
160            "Expected did:keri: prefix, got: {}",
161            did
162        )));
163    }
164
165    let prefix = &did[PREFIX.len()..];
166    if prefix.is_empty() {
167        return Err(ResolveError::InvalidFormat("Empty KERI prefix".into()));
168    }
169
170    // Validate prefix format (starts with E for Blake3 SAID)
171    if !prefix.starts_with('E') {
172        return Err(ResolveError::InvalidFormat(format!(
173            "Invalid KERI prefix format (expected E prefix): {}",
174            prefix
175        )));
176    }
177
178    Ok(Prefix::new_unchecked(prefix.to_string()))
179}
