1//! Git trailer parsing and formatting utilities.
2//!
3//! Git trailers are key-value pairs at the end of a commit message, separated
4//! from the body by a blank line. This module handles:
5//!
6//! - Appending trailers to commit messages
7//! - Parsing trailers (including RFC 822 folded lines)
8//! - Extracting witness receipts from `Auths-Witness-Receipt` trailers
9
10use auths_core::witness::Receipt;
11
12/// The trailer key for witness receipts.
13pub const WITNESS_RECEIPT_KEY: &str = "Auths-Witness-Receipt";
14
15/// Maximum line length before folding (RFC 822 convention).
16const FOLD_WIDTH: usize = 76;
17
18// ── Formatting ──────────────────────────────────────────────────────────────
19
20/// Appends a trailer to a commit message with RFC 822 folding for long values.
21///
22/// If the message doesn't already have a trailer block (blank line before trailers),
23/// a blank line separator is inserted.
24pub fn append_trailer(message: &str, key: &str, value: &str) -> String {
25    let full_line = format!("{}: {}", key, value);
26    let folded = fold_line(&full_line);
27
28    let trimmed = message.trim_end();
29    if trimmed.is_empty() {
30        return folded;
31    }
32
33    // Check if last paragraph looks like trailers already
34    if has_trailer_block(trimmed) {
35        format!("{}\n{}", trimmed, folded)
36    } else {
37        format!("{}\n\n{}", trimmed, folded)
38    }
39}
40
41/// Fold a single line at `FOLD_WIDTH` characters.
42///
43/// Continuation lines start with a single space (RFC 822 convention).
44fn fold_line(line: &str) -> String {
45    if line.len() <= FOLD_WIDTH {
46        return line.to_string();
47    }
48
49    let mut result = String::with_capacity(line.len() + line.len() / FOLD_WIDTH);
50    let mut remaining = line;
51
52    while remaining.len() > FOLD_WIDTH {
53        // Try to break at a space near the fold width, but never at position 0
54        // (position 0 would mean remaining never advances — infinite loop).
55        let break_at = remaining[..FOLD_WIDTH]
56            .rfind(' ')
57            .filter(|&pos| pos > 0)
58            .unwrap_or(FOLD_WIDTH);
59
60        result.push_str(&remaining[..break_at]);
61        result.push('\n');
62        remaining = &remaining[break_at..];
63
64        // Continuation line: ensure it starts with a space
65        if !remaining.starts_with(' ') {
66            result.push(' ');
67        }
68    }
69    result.push_str(remaining);
70    result
71}

73// ── Parsing ─────────────────────────────────────────────────────────────────
74
75/// Parses all trailers from a commit message.
76///
77/// Handles RFC 822 folded lines (continuation lines starting with whitespace).
78/// Returns key-value pairs with folded values rejoined.
79pub fn parse_trailers(message: &str) -> Vec<(String, String)> {
80    let trailer_block = extract_trailer_block(message);
81    if trailer_block.is_empty() {
82        return vec![];
83    }
84
85    let unfolded = unfold_lines(trailer_block);
86    let mut trailers = Vec::new();
87
88    for line in unfolded.lines() {
89        if let Some((key, value)) = parse_trailer_line(line) {
90            trailers.push((key, value));
91        }
92    }
93
94    trailers
95}
96
97/// Extracts and deserializes witness receipts from commit message trailers.
98pub fn extract_witness_receipts(message: &str) -> Vec<Receipt> {
99    parse_trailers(message)
100        .into_iter()
101        .filter(|(key, _)| key == WITNESS_RECEIPT_KEY)
102        .filter_map(|(_, value)| Receipt::from_trailer_value(&value).ok())
103        .collect()
104}

106// ── Internal Helpers ────────────────────────────────────────────────────────

108/// Check if the last paragraph of a message looks like a trailer block.
109fn has_trailer_block(message: &str) -> bool {
110    let last_paragraph = match message.rfind("\n\n") {
111        Some(pos) => &message[pos + 2..],
112        None => message,
113    };
114
115    // A trailer block has at least one line matching "Key: Value"
116    last_paragraph
117        .lines()
118        .any(|line| parse_trailer_line(line).is_some())
119}
120
121/// Extract the trailer block (last paragraph) from a commit message.
122fn extract_trailer_block(message: &str) -> &str {
123    let trimmed = message.trim_end();
124    match trimmed.rfind("\n\n") {
125        Some(pos) => &trimmed[pos + 2..],
126        None => {
127            // Single paragraph — only treat as trailers if every line is a trailer or continuation
128            if trimmed.lines().all(|l| {
129                l.starts_with(' ') || l.starts_with('\t') || parse_trailer_line(l).is_some()
130            }) {
131                trimmed
132            } else {
133                ""
134            }
135        }
136    }
137}

139/// Unfold RFC 822 continuation lines.
140///
141/// Lines starting with whitespace are joined to the previous line.
142fn unfold_lines(text: &str) -> String {
143    let mut result = String::with_capacity(text.len());
144    for line in text.lines() {
145        if line.starts_with(' ') || line.starts_with('\t') {
146            // Continuation — append without newline
147            result.push_str(line);
148        } else {
149            if !result.is_empty() {
150                result.push('\n');
151            }
152            result.push_str(line);
153        }
154    }
155    result
156}
157
158/// Parse a single trailer line into (key, value).
159fn parse_trailer_line(line: &str) -> Option<(String, String)> {
160    let colon_pos = line.find(':')?;
161    let key = line[..colon_pos].trim();
162
163    // Trailer keys must be non-empty, contain no spaces, and be alphanumeric + hyphens
164    if key.is_empty() || key.contains(' ') {
165        return None;
166    }
167    if !key
168        .chars()
169        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
170    {
171        return None;
172    }
173
174    let value = line[colon_pos + 1..].trim().to_string();
175    Some((key.to_string(), value))
176}
