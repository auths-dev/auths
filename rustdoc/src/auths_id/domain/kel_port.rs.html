1use crate::keri::{Event, IcpEvent, KelError, Prefix};

3/// Domain port for Key Event Log operations.
4///
5/// Abstracts how KERI events are stored and retrieved. The domain
6/// provides and receives typed `Event` values without knowledge
7/// of the backing store (Git refs, database, in-memory map).
8///
9/// Args:
10/// * `prefix`: The KERI prefix identifying the identity.
11///
12/// Usage:
13/// ```ignore
14/// use auths_id::domain::kel_port::KelPort;
15/// use auths_id::keri::Prefix;
16///
17/// fn check_identity(kel: &dyn KelPort, prefix: &Prefix) -> bool {
18///     kel.exists(prefix)
19/// }
20/// ```
21pub trait KelPort: Send + Sync {
22    /// Returns whether an event log exists for the given prefix.
23    ///
24    /// Args:
25    /// * `prefix`: The KERI prefix to check.
26    ///
27    /// Usage:
28    /// ```ignore
29    /// if kel.exists(&prefix) {
30    ///     // load events
31    /// }
32    /// ```
33    fn exists(&self, prefix: &Prefix) -> bool;
34
35    /// Returns all events in the log for the given prefix, in order.
36    ///
37    /// Args:
38    /// * `prefix`: The KERI prefix to load events for.
39    ///
40    /// Usage:
41    /// ```ignore
42    /// let events = kel.get_events(&prefix)?;
43    /// ```
44    fn get_events(&self, prefix: &Prefix) -> Result<Vec<Event>, KelError>;
45
46    /// Creates a new event log with an inception event.
47    ///
48    /// Args:
49    /// * `prefix`: The KERI prefix for the new identity.
50    /// * `event`: The inception event to store.
51    ///
52    /// Usage:
53    /// ```ignore
54    /// kel.create(&prefix, &icp_event)?;
55    /// ```
56    fn create(&self, prefix: &Prefix, event: &IcpEvent) -> Result<(), KelError>;
57
58    /// Appends an event to an existing log.
59    ///
60    /// Args:
61    /// * `prefix`: The KERI prefix to append to.
62    /// * `event`: The event to append.
63    ///
64    /// Usage:
65    /// ```ignore
66    /// kel.append(&prefix, &Event::Rot(rot_event))?;
67    /// ```
68    fn append(&self, prefix: &Prefix, event: &Event) -> Result<(), KelError>;
69}
