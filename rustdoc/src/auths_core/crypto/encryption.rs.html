14/// Byte size of the algorithm tag prefix.
15pub const TAG_LEN: usize = 1;
16/// Byte size of the nonce used in both AES-GCM and ChaCha20Poly1305.
17pub const NONCE_LEN: usize = 12; // we're using 12-byte nonces for both
18/// Byte size of the salt used in key derivation.
19pub const SALT_LEN: usize = 16;
20/// Length in bytes of a symmetric encryption key (256-bit).
21pub const SYMMETRIC_KEY_LEN: usize = 32;
22
23/// Tag byte for Argon2id-derived encryption blobs.
24pub const ARGON2_TAG: u8 = 3;
25/// Length of embedded Argon2 parameters: 3 × u32 LE = 12 bytes.
26const ARGON2_PARAMS_LEN: usize = 12;
27
28/// Returns Argon2id parameters for key derivation.
29///
30/// Production builds use OWASP-recommended settings (m=64 MiB, t=3).
31/// Test and `test-utils` builds use minimal settings (m=8 KiB, t=1) to keep
32/// both unit and integration test suites fast. The parameters are embedded in
33/// the encrypted blob, so decryption always uses the values from the blob
34/// header rather than this getter.
35///
36/// Args:
37/// * None
38///
39/// Usage:
40/// ```ignore
41/// let params = get_kdf_params()?;
42/// let argon2 = Argon2::new(Argon2Algorithm::Argon2id, Version::V0x13, params);
43/// ```
44pub fn get_kdf_params() -> Result<Params, AgentError> {
45    #[cfg(not(any(test, feature = "test-utils")))]
46    let params = Params::new(65536, 3, 1, Some(SYMMETRIC_KEY_LEN));
47    #[cfg(any(test, feature = "test-utils"))]
48    let params = Params::new(8, 1, 1, Some(SYMMETRIC_KEY_LEN));
49    params.map_err(|e| AgentError::CryptoError(format!("Invalid Argon2 params: {}", e)))
50}

52/// Validates that a passphrase meets minimum strength requirements.
53///
54/// Requires at least 12 characters and at least 3 of 4 character classes:
55/// lowercase, uppercase, digit, symbol.
56pub fn validate_passphrase(passphrase: &str) -> Result<(), AgentError> {
57    if passphrase.len() < 12 {
58        return Err(AgentError::WeakPassphrase(format!(
59            "Passphrase must be at least 12 characters (got {})",
60            passphrase.len()
61        )));
62    }
63
64    let has_lower = passphrase.chars().any(|c| c.is_ascii_lowercase());
65    let has_upper = passphrase.chars().any(|c| c.is_ascii_uppercase());
66    let has_digit = passphrase.chars().any(|c| c.is_ascii_digit());
67    let has_symbol = passphrase.chars().any(|c| {
68        c.is_ascii_punctuation() || (c.is_ascii() && !c.is_ascii_alphanumeric() && c != ' ')
69    });
70
71    let class_count = has_lower as u8 + has_upper as u8 + has_digit as u8 + has_symbol as u8;
72
73    if class_count < 3 {
74        return Err(AgentError::WeakPassphrase(format!(
75            "Passphrase must contain at least 3 of 4 character classes \
76             (lowercase, uppercase, digit, symbol); found {}",
77            class_count
78        )));
79    }
80
81    Ok(())
82}

84/// Encrypt data using Argon2id for key derivation, prepending tag 0x03.
85///
86/// Output format: `[0x03][salt:16][m_cost:4 LE][t_cost:4 LE][p_cost:4 LE][algo_tag:1][nonce:12][ciphertext]`
87pub fn encrypt_bytes(
88    data: &[u8],
89    passphrase: &str,
90    algo: EncryptionAlgorithm,
91) -> Result<Vec<u8>, AgentError> {
92    validate_passphrase(passphrase)?;
93
94    let salt: [u8; SALT_LEN] = rand::random();
95
96    let params = get_kdf_params()?;
97    let m_cost = params.m_cost();
98    let t_cost = params.t_cost();
99    let p_cost = params.p_cost();
100    let argon2 = Argon2::new(Argon2Algorithm::Argon2id, Version::V0x13, params);
101    let mut key = Zeroizing::new([0u8; SYMMETRIC_KEY_LEN]);
102    argon2
103        .hash_password_into(passphrase.as_bytes(), &salt, &mut *key)
104        .map_err(|e| AgentError::CryptoError(format!("Argon2 key derivation failed: {}", e)))?;
105
106    let nonce: [u8; NONCE_LEN] = rand::random();
107
108    let ciphertext = match algo {
109        EncryptionAlgorithm::AesGcm256 => {
110            let cipher = Aes256Gcm::new_from_slice(&*key)
111                .map_err(|_| AgentError::CryptoError("Invalid AES key".into()))?;
112            cipher
113                .encrypt(AesNonce::from_slice(&nonce), data)
114                .map_err(|_| AgentError::CryptoError("AES encryption failed".into()))?
115        }
116        EncryptionAlgorithm::ChaCha20Poly1305 => {
117            let cipher = ChaCha20Poly1305::new_from_slice(&*key)
118                .map_err(|_| AgentError::CryptoError("Invalid ChaCha key".into()))?;
119            cipher
120                .encrypt(ChaChaNonce::from_slice(&nonce), data)
121                .map_err(|_| AgentError::CryptoError("ChaCha encryption failed".into()))?
122        }
123    };
124
125    let mut out = Vec::with_capacity(
126        TAG_LEN + SALT_LEN + ARGON2_PARAMS_LEN + TAG_LEN + NONCE_LEN + ciphertext.len(),
127    );
128    out.push(ARGON2_TAG);
129    out.extend_from_slice(&salt);
130    out.extend_from_slice(&m_cost.to_le_bytes());
131    out.extend_from_slice(&t_cost.to_le_bytes());
132    out.extend_from_slice(&p_cost.to_le_bytes());
133    out.push(algo.tag());
134    out.extend_from_slice(&nonce);
135    out.extend_from_slice(&ciphertext);
136    Ok(out)
137}

139/// Decrypts data using a tagged encryption format and a user-provided passphrase.
140///
141/// Only supports tag 3 (Argon2id): `[0x03][salt:16][m_cost:4 LE][t_cost:4 LE][p_cost:4 LE][algo_tag:1][nonce:12][ciphertext]`
142///
143/// If decryption fails (e.g. due to wrong passphrase), returns
144/// `AgentError::IncorrectPassphrase`.
145pub fn decrypt_bytes(encrypted: &[u8], passphrase: &str) -> Result<Vec<u8>, AgentError> {
146    if encrypted.is_empty() {
147        return Err(AgentError::CryptoError("Encrypted data too short".into()));
148    }
149
150    let tag = encrypted[0];
151
152    if tag == ARGON2_TAG {
153        return decrypt_bytes_argon2(encrypted, passphrase);
154    }
155
156    // Legacy HKDF tags 1 (AES) and 2 (ChaCha) are no longer supported
157    if tag == 1 || tag == 2 {
158        return Err(AgentError::CryptoError(
159            "This key was encrypted with a legacy format (HKDF). \
160             Re-encrypt it with: auths key migrate"
161                .into(),
162        ));
163    }
164
165    Err(AgentError::CryptoError(format!(
166        "Unknown encryption tag: {}",
167        tag
168    )))
169}

171/// Decrypts an Argon2id-tagged blob.
172///
173/// Expected format: `[0x03][salt:16][m_cost:4 LE][t_cost:4 LE][p_cost:4 LE][algo_tag:1][nonce:12][ciphertext]`
174fn decrypt_bytes_argon2(encrypted: &[u8], passphrase: &str) -> Result<Vec<u8>, AgentError> {
175    const MIN_LEN: usize = TAG_LEN + SALT_LEN + ARGON2_PARAMS_LEN + TAG_LEN + NONCE_LEN;
176    if encrypted.len() < MIN_LEN {
177        return Err(AgentError::CryptoError(
178            "Argon2id encrypted data too short".into(),
179        ));
180    }
181
182    let mut offset = TAG_LEN; // skip the 0x03 tag
183
184    let salt = &encrypted[offset..offset + SALT_LEN];
185    offset += SALT_LEN;
186
187    let m_cost = u32::from_le_bytes(
188        encrypted[offset..offset + 4]
189            .try_into()
190            .map_err(|_| AgentError::CryptoError("invalid m_cost bytes".into()))?,
191    );
192    offset += 4;
193    let t_cost = u32::from_le_bytes(
194        encrypted[offset..offset + 4]
195            .try_into()
196            .map_err(|_| AgentError::CryptoError("invalid t_cost bytes".into()))?,
197    );
198    offset += 4;
199    let p_cost = u32::from_le_bytes(
200        encrypted[offset..offset + 4]
201            .try_into()
202            .map_err(|_| AgentError::CryptoError("invalid p_cost bytes".into()))?,
203    );
204    offset += 4;
205
206    let algo_tag = encrypted[offset];
207    offset += 1;
208    let algo = EncryptionAlgorithm::from_tag(algo_tag)
209        .ok_or_else(|| AgentError::CryptoError(format!("Unknown encryption tag: {}", algo_tag)))?;
210
211    let nonce = &encrypted[offset..offset + NONCE_LEN];
212    offset += NONCE_LEN;
213
214    let ciphertext = &encrypted[offset..];
215
216    let params = Params::new(m_cost, t_cost, p_cost, Some(SYMMETRIC_KEY_LEN))
217        .map_err(|e| AgentError::CryptoError(format!("Invalid Argon2 params: {}", e)))?;
218    let argon2 = Argon2::new(Argon2Algorithm::Argon2id, Version::V0x13, params);
219    let mut key = Zeroizing::new([0u8; SYMMETRIC_KEY_LEN]);
220    argon2
221        .hash_password_into(passphrase.as_bytes(), salt, &mut *key)
222        .map_err(|e| AgentError::CryptoError(format!("Argon2 key derivation failed: {}", e)))?;
223
224    let result = match algo {
225        EncryptionAlgorithm::AesGcm256 => Aes256Gcm::new_from_slice(&*key)
226            .map_err(|_| AgentError::CryptoError("Invalid AES key".into()))?
227            .decrypt(AesNonce::from_slice(nonce), ciphertext),
228
229        EncryptionAlgorithm::ChaCha20Poly1305 => ChaCha20Poly1305::new_from_slice(&*key)
230            .map_err(|_| AgentError::CryptoError("Invalid ChaCha key".into()))?
231            .decrypt(ChaChaNonce::from_slice(nonce), ciphertext),
232    };
233
234    match result {
235        Ok(plaintext) => Ok(plaintext),
236        Err(_) => Err(AgentError::IncorrectPassphrase),
237    }
238}
