1//! SAID (Self-Addressing Identifier) computation for KERI.
2//!
3//! This module provides functions for computing SAIDs and next-key commitments
4//! as specified by KERI (Key Event Receipt Infrastructure).
5//!
6//! SAIDs use Blake3 hashing with Base64url encoding and an 'E' prefix
7//! (derivation code for Blake3-256).
8
9use auths_verifier::keri::Said;
10use base64::{Engine, engine::general_purpose::URL_SAFE_NO_PAD};
11use subtle::ConstantTimeEq;
12
13/// Compute SAID (Self-Addressing Identifier) for a KERI event.
14///
15// SYNC: must match auths-verifier/src/keri.rs — tested by said_cross_validation
16/// The SAID is computed by:
17/// 1. Hashing the input with Blake3
18/// 2. Encoding the hash as Base64url (no padding)
19/// 3. Prefixing with 'E' (KERI derivation code for Blake3-256)
20///
21/// # Arguments
22/// * `event_json` - The canonical JSON bytes of the event (with 'd' field empty)
23///
24/// # Returns
25/// A `Said` wrapping a string like "EXq5YqaL6L48pf0fu7IUhL0JRaU2_RxFP0AL43wYn148"
26pub fn compute_said(event_json: &[u8]) -> Said {
27    let hash = blake3::hash(event_json);
28    let encoded = URL_SAFE_NO_PAD.encode(hash.as_bytes());
29    Said::new_unchecked(format!("E{}", encoded))
30}
31
32/// Compute next-key commitment hash for pre-rotation.
33///
34// SYNC: must match auths-verifier/src/keri.rs — tested by said_cross_validation
35/// The commitment is computed by:
36/// 1. Hashing the public key bytes with Blake3
37/// 2. Encoding the hash as Base64url (no padding)
38/// 3. Prefixing with 'E' (KERI derivation code for Blake3-256)
39///
40/// This commitment is included in the current event's 'n' field and must
41/// be satisfied by the next rotation event's 'k' field.
42///
43/// # Arguments
44/// * `public_key` - The raw public key bytes (32 bytes for Ed25519)
45///
46/// # Returns
47/// A commitment string like "EO8CE5RH3wHBrXyFay3MOXq5YqaL6L48pf0fu7IUhL0J"
48pub fn compute_next_commitment(public_key: &[u8]) -> String {
49    let hash = blake3::hash(public_key);
50    let encoded = URL_SAFE_NO_PAD.encode(hash.as_bytes());
51    format!("E{}", encoded)
52}
53
54/// Verify that a public key matches a commitment.
55///
56/// # Arguments
57/// * `public_key` - The raw public key bytes to verify
58/// * `commitment` - The commitment string from a previous event's 'n' field
59///
60/// # Returns
61/// `true` if the public key hashes to the commitment, `false` otherwise
62// Defense-in-depth: both values are derived from public data, but constant-time
63// comparison prevents timing side-channels on commitment verification.
64pub fn verify_commitment(public_key: &[u8], commitment: &str) -> bool {
65    let computed = compute_next_commitment(public_key);
66    computed.as_bytes().ct_eq(commitment.as_bytes()).into()
67}

69#[cfg(test)]
70mod tests {
71    use super::*;
72
73    #[test]
74    fn said_is_deterministic() {
75        let json = b"{\"t\":\"icp\",\"s\":\"0\"}";
76        let said1 = compute_said(json);
77        let said2 = compute_said(json);
78        assert_eq!(said1, said2);
79        assert!(said1.as_str().starts_with('E'));
80    }
81
82    #[test]
83    fn said_has_correct_length() {
84        let json = b"{\"test\":\"data\"}";
85        let said = compute_said(json);
86        // 'E' + 43 chars of base64url (32 bytes encoded)
87        assert_eq!(said.as_str().len(), 44);
88    }
89
90    #[test]
91    fn different_inputs_produce_different_saids() {
92        let said1 = compute_said(b"{\"a\":1}");
93        let said2 = compute_said(b"{\"a\":2}");
94        assert_ne!(said1, said2);
95    }
96
97    #[test]
98    fn commitment_verification_works() {
99        let key = [1u8; 32];
100        let commitment = compute_next_commitment(&key);
101        assert!(verify_commitment(&key, &commitment));
102        assert!(!verify_commitment(&[2u8; 32], &commitment));
103    }
104
105    #[test]
106    fn commitment_is_deterministic() {
107        let key = [42u8; 32];
108        let c1 = compute_next_commitment(&key);
109        let c2 = compute_next_commitment(&key);
110        assert_eq!(c1, c2);
111        assert!(c1.starts_with('E'));
112    }
113
114    #[test]
115    fn commitment_has_correct_length() {
116        let key = [0u8; 32];
117        let commitment = compute_next_commitment(&key);
118        // 'E' + 43 chars of base64url
119        assert_eq!(commitment.len(), 44);
120    }
