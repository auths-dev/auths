9/// Extract an Ed25519 seed from PKCS#8 key bytes.
10///
11/// Delegates to `auths_crypto::parse_ed25519_seed` and wraps the result
12/// in the SSH-specific `CryptoError`.
13///
14/// Args:
15/// * `pkcs8`: PKCS#8 encoded key material (v1 or v2).
16///
17/// Usage:
18/// ```ignore
19/// let seed = extract_seed_from_pkcs8(&pkcs8)?;
20/// let sshsig = create_sshsig(&seed, data, "git")?;
21/// ```
22pub fn extract_seed_from_pkcs8(pkcs8: &Pkcs8Der) -> Result<SecureSeed, CryptoError> {
23    auths_crypto::parse_ed25519_seed(pkcs8.as_ref()).map_err(CryptoError::from)
24}
25
26/// Build a PKCS#8 v2 DER document from a seed, deriving the public key internally.
27///
28/// Args:
29/// * `seed`: The 32-byte Ed25519 seed.
30///
31/// Usage:
32/// ```ignore
33/// let pkcs8 = build_ed25519_pkcs8_v2_from_seed(&seed)?;
34/// ```
35pub fn build_ed25519_pkcs8_v2_from_seed(seed: &SecureSeed) -> Result<Pkcs8Der, CryptoError> {
36    let ssh_kp = SshEd25519Keypair::from_seed(seed.as_bytes());
37    let pubkey: [u8; 32] = ssh_kp.public.0;
38    let pkcs8 = build_ed25519_pkcs8_v2(seed.as_bytes(), &pubkey);
39    Ok(Pkcs8Der::new(pkcs8))
40}

42/// Extract the 32-byte Ed25519 public key from key bytes.
43///
44/// Parses key material to find the embedded public key (PKCS#8 v2), or
45/// derives it from the seed when only PKCS#8 v1 or raw bytes are provided.
46///
47/// Args:
48/// * `key_bytes`: Key material in any supported PKCS#8 or raw format.
49///
50/// Usage:
51/// ```ignore
52/// let pubkey = extract_pubkey_from_key_bytes(pkcs8.as_ref())?;
53/// ```
54pub fn extract_pubkey_from_key_bytes(key_bytes: &[u8]) -> Result<Vec<u8>, CryptoError> {
55    let (seed, maybe_pubkey) = parse_ed25519_key_material(key_bytes).map_err(CryptoError::from)?;
56
57    match maybe_pubkey {
58        Some(pk) => Ok(pk.to_vec()),
59        None => {
60            let ssh_kp = SshEd25519Keypair::from_seed(seed.as_bytes());
61            Ok(ssh_kp.public.0.to_vec())
62        }
63    }
64}
