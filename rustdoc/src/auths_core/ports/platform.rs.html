10/// Errors from platform identity claim operations (OAuth, proof publishing, registry submission).
11///
12/// Usage:
13/// ```ignore
14/// match result {
15///     Err(PlatformError::AccessDenied) => { /* user denied */ }
16///     Err(PlatformError::ExpiredToken) => { /* device code expired, restart flow */ }
17///     Err(PlatformError::Network(e)) => { /* retry */ }
18///     Err(e) => return Err(e.into()),
19///     Ok(response) => { /* proceed */ }
20/// }
21/// ```
22#[derive(Debug, Error)]
23#[non_exhaustive]
24pub enum PlatformError {
25    /// OAuth authorization is pending — the user has not yet approved.
26    #[error("OAuth authorization pending")]
27    AuthorizationPending,
28    /// OAuth server requested slower polling.
29    #[error("OAuth slow down")]
30    SlowDown,
31    /// The user explicitly denied the OAuth authorization request.
32    #[error("OAuth access denied")]
33    AccessDenied,
34    /// The device code has expired; the flow must be restarted.
35    #[error("device code expired")]
36    ExpiredToken,
37    /// A network-level error occurred.
38    #[error("network error: {0}")]
39    Network(#[source] NetworkError),
40    /// A platform-specific error with a human-readable message.
41    #[error("platform error: {message}")]
42    Platform {
43        /// Human-readable error detail from the platform API.
44        message: String,
45    },
46}

48/// OAuth 2.0 device authorization grant response (RFC 8628 §3.2).
49///
50/// Returned by [`OAuthDeviceFlowProvider::request_device_code`].
51/// The CLI displays `user_code` + `verification_uri` to the user,
52/// then polls with `device_code`.
53pub struct DeviceCodeResponse {
54    /// Opaque device verification code used to poll for the token.
55    pub device_code: String,
56    /// Short user-facing code to enter at `verification_uri`.
57    pub user_code: String,
58    /// URL where the user enters `user_code`.
59    pub verification_uri: String,
60    /// Lifetime of the device code in seconds.
61    pub expires_in: u64,
62    /// Minimum polling interval in seconds.
63    pub interval: u64,
64}

84/// Two-phase OAuth 2.0 device authorization flow (RFC 8628).
85///
86/// The CLI calls [`request_device_code`](Self::request_device_code), displays the
87/// `user_code` and `verification_uri`, optionally opens a browser, then calls
88/// [`poll_for_token`](Self::poll_for_token). All presentation logic stays in the CLI.
89///
90/// Usage:
91/// ```ignore
92/// let resp = provider.request_device_code(CLIENT_ID, "read:user gist").await?;
93/// // CLI: display resp.user_code, open resp.verification_uri
94/// let token = provider.poll_for_token(CLIENT_ID, &resp.device_code, Duration::from_secs(resp.interval), expires_at).await?;
95/// let profile = provider.fetch_user_profile(&token).await?;
96/// ```
97pub trait OAuthDeviceFlowProvider: Send + Sync {
98    /// Request a device code to begin the device authorization flow.
99    ///
100    /// Args:
101    /// * `client_id`: OAuth application client ID.
102    /// * `scopes`: Space-separated OAuth scopes to request.
103    fn request_device_code(
104        &self,
105        client_id: &str,
106        scopes: &str,
107    ) -> impl Future<Output = Result<DeviceCodeResponse, PlatformError>> + Send;
108
109    /// Poll for the access token until granted, denied, or the lifetime elapses.
110    ///
111    /// Args:
112    /// * `client_id`: OAuth application client ID.
113    /// * `device_code`: Device code from [`request_device_code`](Self::request_device_code).
114    /// * `interval`: Minimum time between poll attempts.
115    /// * `expires_in`: Total lifetime of the device code (stop polling after this elapses).
116    fn poll_for_token(
117        &self,
118        client_id: &str,
119        device_code: &str,
120        interval: Duration,
121        expires_in: Duration,
122    ) -> impl Future<Output = Result<String, PlatformError>> + Send;
123
124    /// Fetch the authenticated user's profile using the access token.
125    ///
126    /// Args:
127    /// * `access_token`: OAuth access token from [`poll_for_token`](Self::poll_for_token).
128    fn fetch_user_profile(
129        &self,
130        access_token: &str,
131    ) -> impl Future<Output = Result<PlatformUserProfile, PlatformError>> + Send;
132}

134/// Publish a signed platform claim as a publicly readable proof artifact.
135///
136/// For GitHub: publishes a Gist. Returns the URL of the published proof.
137///
138/// Usage:
139/// ```ignore
140/// let proof_url = publisher.publish_proof(&access_token, &claim_json).await?;
141/// registry_client.submit_claim(registry, &did, &proof_url).await?;
142/// ```
143pub trait PlatformProofPublisher: Send + Sync {
144    /// Publish the claim JSON as a proof artifact and return its public URL.
145    ///
146    /// Args:
147    /// * `access_token`: OAuth access token with write permission to publish.
148    /// * `claim_json`: Canonicalized, signed JSON claim to publish.
149    fn publish_proof(
150        &self,
151        access_token: &str,
152        claim_json: &str,
153    ) -> impl Future<Output = Result<String, PlatformError>> + Send;
154}

156/// Submit a published platform claim to the auths registry for verification.
157///
158/// Usage:
159/// ```ignore
160/// let response = registry_client.submit_claim(registry_url, &did, &proof_url).await?;
161/// println!("{}", response.message);
162/// ```
163pub trait RegistryClaimClient: Send + Sync {
164    /// Submit a platform identity claim to the registry.
165    ///
166    /// Args:
167    /// * `registry_url`: Base URL of the auths registry.
168    /// * `did`: DID of the identity making the claim.
169    /// * `proof_url`: Public URL of the published proof artifact.
170    fn submit_claim(
171        &self,
172        registry_url: &str,
173        did: &str,
174        proof_url: &str,
175    ) -> impl Future<Output = Result<ClaimResponse, PlatformError>> + Send;
176}
