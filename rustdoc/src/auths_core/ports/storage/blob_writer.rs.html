3/// Writes and deletes arbitrary binary blobs in storage by logical path.
4///
5/// Implementations handle the details of persisting bytes to the backing
6/// store (Git tree mutations, filesystem writes, database inserts).
7///
8/// Usage:
9/// ```ignore
10/// use auths_core::ports::storage::BlobWriter;
11///
12/// fn save_metadata(writer: &dyn BlobWriter, id: &str, data: &[u8]) {
13///     let path = format!("identities/{id}/metadata");
14///     writer.put_blob(&path, data).unwrap();
15/// }
16/// ```
17pub trait BlobWriter: Send + Sync {
18    /// Writes raw bytes to the given logical path, creating or overwriting.
19    ///
20    /// Args:
21    /// * `path`: The logical storage path for the blob.
22    /// * `data`: The raw bytes to store.
23    ///
24    /// Usage:
25    /// ```ignore
26    /// writer.put_blob("identities/abc123/metadata", &serialized)?;
27    /// ```
28    fn put_blob(&self, path: &str, data: &[u8]) -> Result<(), StorageError>;
29
30    /// Deletes the blob at the given logical path.
31    ///
32    /// Args:
33    /// * `path`: The logical storage path to delete.
34    ///
35    /// Usage:
36    /// ```ignore
37    /// writer.delete_blob("identities/abc123/metadata")?;
38    /// ```
39    fn delete_blob(&self, path: &str) -> Result<(), StorageError>;
40}
