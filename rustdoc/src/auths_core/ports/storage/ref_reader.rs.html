3/// Reads abstract references that map names to opaque identifiers.
4///
5/// In a Git-backed implementation, these correspond to Git refs pointing
6/// at commit OIDs. The domain treats them as opaque byte sequences
7/// without assuming Git semantics.
8///
9/// Usage:
10/// ```ignore
11/// use auths_core::ports::storage::RefReader;
12///
13/// fn current_tip(reader: &dyn RefReader, refname: &str) -> Vec<u8> {
14///     reader.resolve_ref(refname).unwrap()
15/// }
16/// ```
17pub trait RefReader: Send + Sync {
18    /// Resolves a reference name to its current target bytes.
19    ///
20    /// Args:
21    /// * `refname`: The reference name to resolve (e.g., `"refs/auths/registry"`).
22    ///
23    /// Usage:
24    /// ```ignore
25    /// let oid_bytes = reader.resolve_ref("refs/auths/registry")?;
26    /// ```
27    fn resolve_ref(&self, refname: &str) -> Result<Vec<u8>, StorageError>;
28
29    /// Lists all reference names matching a glob pattern.
30    ///
31    /// Args:
32    /// * `glob`: A glob pattern to match against reference names.
33    ///
34    /// Usage:
35    /// ```ignore
36    /// let refs = reader.list_refs("refs/auths/devices/nodes/*/signatures")?;
37    /// ```
38    fn list_refs(&self, glob: &str) -> Result<Vec<String>, StorageError>;
39}
