1//! Organization authorization policy.
2//!
3//! This module implements org membership authorization rules that determine
4//! whether a member can perform actions on behalf of an organization.
5//!
6//! # Org Membership vs Device Authorization
7//!
8//! - **Device authorization** ([`super::device`]): Can this device act for an identity?
9//! - **Org authorization** (this module): Can this member act for an organization?
10//!
11//! Both use attestations but with different issuers:
12//! - Device attestations: issued by `did:keri:{identity_prefix}`
13//! - Org membership attestations: issued by `did:keri:{org_prefix}`
14//!
15//! # Integration with MemberFilter
16//!
17//! This policy evaluates individual membership attestations. For bulk queries
18//! with filtering (role, capabilities), use the registry's `list_org_members()`
19//! with `MemberFilter`, then apply this policy to each result.
20

28/// Authorize an org member to perform an action.
29///
30/// # Sans-IO Design
31///
32/// All inputs are passed explicitly:
33/// - No storage access (membership attestation provided by caller)
34/// - No system clock (time injected via `now`)
35/// - Pure function: same inputs always produce same output
36///
37/// # Rules (evaluated in order)
38///
39/// 1. **Not revoked**: `!att.is_revoked()`
40/// 2. **Not expired**: `att.expires_at > now` OR `att.expires_at.is_none()`
41/// 3. **Issuer is org**: `att.issuer == expected_org_issuer`
42/// 4. **Capability allows action**: `action.to_capability() in att.capabilities`
43///
44/// # Arguments
45///
46/// * `member_attestation` - The member's org membership attestation
47/// * `expected_org_issuer` - The expected org issuer DID (e.g., `did:keri:EOrg...`)
48/// * `action` - The action the member wants to perform
49/// * `now` - Current time for expiry checks
50///
51/// # Returns
52///
53/// `Decision::Allow` if all rules pass, `Decision::Deny` otherwise.
54///
55/// # Examples
56///
57/// ```rust
58/// use auths_core::policy::{Decision, device::Action, org::authorize_org_action};
59/// use auths_verifier::core::{Attestation, Capability, Ed25519PublicKey, Ed25519Signature, Role};
60/// use auths_verifier::types::DeviceDID;
61/// use chrono::Utc;
62///
63/// let membership = Attestation {
64///     version: 1,
65///     rid: "member".into(),
66///     issuer: "did:keri:EOrg123".into(),
67///     subject: DeviceDID::new("did:key:z6MkAlice"),
68///     device_public_key: Ed25519PublicKey::from_bytes([0u8; 32]),
69///     identity_signature: Ed25519Signature::empty(),
70///     device_signature: Ed25519Signature::empty(),
71///     revoked_at: None,
72///     expires_at: None,
73///     timestamp: None,
74///     note: None,
75///     payload: None,
76///     role: Some(Role::Admin),
77///     capabilities: vec![Capability::manage_members()],
78///     delegated_by: None,
79///     signer_type: None,
80///     environment_claim: None,
81/// };
82///
83/// let decision = authorize_org_action(
84///     &membership,
85///     "did:keri:EOrg123",
86///     &Action::ManageMembers,
87///     Utc::now(),
88/// );
89///
90/// assert!(decision.is_allowed());
91/// ```
92pub fn authorize_org_action(
93    member_attestation: &Attestation,
94    expected_org_issuer: &str,
95    action: &Action,
96    now: DateTime<Utc>,
97) -> Decision {
98    // Rule 1: Not revoked
99    if member_attestation.is_revoked() {
100        return Decision::deny("membership is revoked");
101    }
102
103    // Rule 2: Not expired (expires_at <= now means expired)
104    if let Some(expires_at) = member_attestation.expires_at
105        && expires_at <= now
106    {
107        return Decision::deny(format!(
108            "membership expired at {}",
109            expires_at.format("%Y-%m-%dT%H:%M:%SZ")
110        ));
111    }
112
113    // Rule 3: Issuer is the org
114    if member_attestation.issuer != expected_org_issuer {
115        return Decision::deny(format!(
116            "issuer mismatch: expected org '{}', got '{}'",
117            expected_org_issuer, member_attestation.issuer
118        ));
119    }
120
121    // Rule 4: Capability allows action
122    let required_capability = match action.to_capability() {
123        Ok(cap) => cap,
124        Err(msg) => return Decision::deny(msg),
125    };
126
127    // Empty capabilities means no permissions
128    if member_attestation.capabilities.is_empty() {
129        return Decision::deny("membership has no capabilities");
130    }
131
132    if !member_attestation
133        .capabilities
134        .contains(&required_capability)
135    {
136        return Decision::deny(format!(
137            "capability '{}' not granted by membership",
138            capability_name(&required_capability)
139        ));
140    }
141
142    // All rules passed
143    Decision::allow(format!(
144        "member authorized for '{}' in org",
145        capability_name(&required_capability)
146    ))
147}

149/// Derive the expected issuer DID for an org prefix.
150///
151/// Org membership attestations must be issued by the org identity itself.
152///
153/// # Examples
154///
155/// ```rust
156/// use auths_core::policy::org::expected_org_issuer;
157/// use auths_verifier::keri::Prefix;
158///
159/// let prefix = Prefix::new_unchecked("EOrg12345".into());
160/// assert_eq!(expected_org_issuer(&prefix), "did:keri:EOrg12345");
161/// ```
162pub fn expected_org_issuer(org_prefix: &Prefix) -> String {
163    format!("did:keri:{}", org_prefix.as_str())
164}

166/// Get a human-readable name for a capability.
167fn capability_name(cap: &Capability) -> &str {
168    cap.as_str()
169}

179    fn make_membership(
180        revoked_at: Option<DateTime<Utc>>,
181        expires_at: Option<DateTime<Utc>>,
182        issuer: &str,
183        capabilities: Vec<Capability>,
184        role: Option<Role>,
185    ) -> Attestation {
186        Attestation {
187            version: 1,
188            rid: ResourceId::new("membership"),
189            issuer: issuer.into(),
190            subject: DeviceDID::new("did:key:z6MkMember"),
191            device_public_key: Ed25519PublicKey::from_bytes([0u8; 32]),
192            identity_signature: Ed25519Signature::empty(),
193            device_signature: Ed25519Signature::empty(),
194            revoked_at,
195            expires_at,
196            timestamp: None,
197            note: None,
198            payload: None,
199            role,
200            capabilities,
201            delegated_by: None,
202            signer_type: None,
203            environment_claim: None,
204        }
205    }

209    #[test]
210    fn valid_membership_allows() {
211        let att = make_membership(
212            None,
213            None,
214            ORG_ISSUER,
215            vec![Capability::manage_members()],
216            Some(Role::Admin),
217        );
218        let now = Utc::now();
219
220        let decision = authorize_org_action(&att, ORG_ISSUER, &Action::ManageMembers, now);
221
222        assert!(decision.is_allowed());
223        assert!(decision.reason().contains("authorized"));
224    }
