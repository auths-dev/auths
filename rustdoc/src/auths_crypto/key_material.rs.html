1//! Ed25519 key material parsing from various serialization formats.
2//!
3//! Extracts [`SecureSeed`] (and optionally the public key) from PKCS#8 v1, v2,
4//! raw 32-byte seeds, and OCTET-STRING-wrapped seeds — pure byte parsing with
5//! no backend dependency.
6
7use crate::provider::{CryptoError, SecureSeed};
8
9/// PKCS#8 v2 Ed25519 total length — explicit [1] tag (85 bytes).
10const PKCS8_V2_EXPLICIT_LEN: usize = 85;
11
12/// PKCS#8 v2 Ed25519 total length — implicit [1] tag, as produced by ring (83 bytes).
13const PKCS8_V2_IMPLICIT_LEN: usize = 83;
14
15/// PKCS#8 v1 Ed25519 total length (48 bytes) — seed wrapped in inner OCTET STRING.
16const PKCS8_V1_LEN: usize = 48;
17
18/// PKCS#8 v1 Ed25519 with unwrapped seed (46 bytes) — seed directly in outer OCTET STRING.
19const PKCS8_V1_UNWRAPPED_LEN: usize = 46;
20
21/// Offset where the 32-byte seed starts in PKCS#8 formats with inner OCTET STRING wrapper.
22const SEED_OFFSET: usize = 16;
23
24/// Offset where the 32-byte seed starts in PKCS#8 v1 without inner OCTET STRING wrapper.
25const SEED_OFFSET_UNWRAPPED: usize = 14;
26
27/// Offset where the 32-byte public key starts in PKCS#8 v2 (explicit [1] tag).
28const PUBKEY_OFFSET_EXPLICIT: usize = 53;
29
30/// Offset where the 32-byte public key starts in PKCS#8 v2 (implicit [1] tag, ring format).
31const PUBKEY_OFFSET_IMPLICIT: usize = 51;
32
33/// Parse an Ed25519 seed from key bytes in various formats.
34///
35/// Handles:
36/// 1. PKCS#8 v2 explicit tag (85 bytes) — seed at bytes [16..48]
37/// 2. PKCS#8 v2 implicit tag / ring (83 bytes) — seed at bytes [16..48]
38/// 3. PKCS#8 v1 wrapped (48 bytes) — seed at bytes [16..48]
39/// 4. PKCS#8 v1 unwrapped (46 bytes) — seed at bytes [14..46]
40/// 5. Raw 32-byte seed
41/// 6. Raw 34-byte OCTET STRING wrapped seed (04 20 prefix)
42///
43/// Args:
44/// * `bytes`: Key material in any supported format.
45///
46/// Usage:
47/// ```ignore
48/// let seed = parse_ed25519_seed(&pkcs8_bytes)?;
49/// let sig = provider.sign_ed25519(&seed, b"msg").await?;
50/// ```
51pub fn parse_ed25519_seed(bytes: &[u8]) -> Result<SecureSeed, CryptoError> {
52    match bytes.len() {
53        PKCS8_V2_EXPLICIT_LEN | PKCS8_V2_IMPLICIT_LEN | PKCS8_V1_LEN => {
54            extract_seed_at(bytes, SEED_OFFSET)
55        }
56        PKCS8_V1_UNWRAPPED_LEN => extract_seed_at(bytes, SEED_OFFSET_UNWRAPPED),
57        32 => {
58            let mut buf = [0u8; 32];
59            buf.copy_from_slice(bytes);
60            Ok(SecureSeed::new(buf))
61        }
62        34 if bytes[0] == 0x04 && bytes[1] == 0x20 => {
63            let mut buf = [0u8; 32];
64            buf.copy_from_slice(&bytes[2..]);
65            Ok(SecureSeed::new(buf))
66        }
67        _ => Err(CryptoError::InvalidPrivateKey(format!(
68            "Unrecognized Ed25519 key format ({} bytes)",
69            bytes.len()
70        ))),
71    }
72}

74/// Parse an Ed25519 seed and, when available, the public key from key bytes.
75///
76/// Returns `(seed, Some(pubkey))` for PKCS#8 v2 (which embeds the public key),
77/// or `(seed, None)` for formats that don't include one. Callers can derive the
78/// public key from the seed via `CryptoProvider::ed25519_public_key_from_seed`.
79///
80/// Args:
81/// * `bytes`: Key material in any supported format.
82///
83/// Usage:
84/// ```ignore
85/// let (seed, maybe_pk) = parse_ed25519_key_material(&pkcs8_bytes)?;
86/// let pk = match maybe_pk {
87///     Some(pk) => pk,
88///     None => provider.ed25519_public_key_from_seed(&seed).await?,
89/// };
90/// ```
91pub fn parse_ed25519_key_material(
92    bytes: &[u8],
93) -> Result<(SecureSeed, Option<[u8; 32]>), CryptoError> {
94    let seed = parse_ed25519_seed(bytes)?;
95
96    let pubkey = match bytes.len() {
97        PKCS8_V2_EXPLICIT_LEN => {
98            let mut pk = [0u8; 32];
99            pk.copy_from_slice(&bytes[PUBKEY_OFFSET_EXPLICIT..PUBKEY_OFFSET_EXPLICIT + 32]);
100            Some(pk)
101        }
102        PKCS8_V2_IMPLICIT_LEN => {
103            let mut pk = [0u8; 32];
104            pk.copy_from_slice(&bytes[PUBKEY_OFFSET_IMPLICIT..PUBKEY_OFFSET_IMPLICIT + 32]);
105            Some(pk)
106        }
107        _ => None,
108    };
109
110    Ok((seed, pubkey))
111}

113/// Build a PKCS#8 v2 DER encoding from a raw seed and public key.
114///
115/// Produces an 85-byte document (explicit \[1\] tag) compatible with ring's
116/// `Ed25519KeyPair::from_pkcs8` (which accepts both 83 and 85-byte forms).
117///
118/// Args:
119/// * `seed`: Raw 32-byte Ed25519 private key seed.
120/// * `pubkey`: Raw 32-byte Ed25519 public key.
121///
122/// Usage:
123/// ```ignore
124/// let (seed, pk) = provider.generate_ed25519_keypair().await?;
125/// let pkcs8 = build_ed25519_pkcs8_v2(seed.as_bytes(), &pk);
126/// assert_eq!(pkcs8.len(), 85);
127/// ```
128pub fn build_ed25519_pkcs8_v2(seed: &[u8; 32], pubkey: &[u8; 32]) -> Vec<u8> {
129    let mut buf = Vec::with_capacity(PKCS8_V2_EXPLICIT_LEN);
130    // SEQUENCE (83 bytes payload)
131    buf.extend_from_slice(&[0x30, 0x53]);
132    // INTEGER 1 (version = v2)
133    buf.extend_from_slice(&[0x02, 0x01, 0x01]);
134    // SEQUENCE { OID 1.3.101.112 }
135    buf.extend_from_slice(&[0x30, 0x05, 0x06, 0x03, 0x2b, 0x65, 0x70]);
136    // OCTET STRING (34 bytes) containing OCTET STRING (32 bytes) containing seed
137    buf.extend_from_slice(&[0x04, 0x22, 0x04, 0x20]);
138    buf.extend_from_slice(seed);
139    // [1] EXPLICIT BIT STRING (33 bytes: 0x00 pad + 32 bytes pubkey)
140    buf.extend_from_slice(&[0xa1, 0x23, 0x03, 0x21, 0x00]);
141    buf.extend_from_slice(pubkey);
142    buf
143}
144
145fn extract_seed_at(bytes: &[u8], offset: usize) -> Result<SecureSeed, CryptoError> {
146    if bytes.len() < offset + 32 {
147        return Err(CryptoError::InvalidPrivateKey(
148            "Key bytes too short for seed extraction".to_string(),
149        ));
150    }
151    let mut buf = [0u8; 32];
152    buf.copy_from_slice(&bytes[offset..offset + 32]);
153    Ok(SecureSeed::new(buf))
154}

156#[cfg(test)]
157mod tests {
158    use super::*;
159
160    #[test]
161    fn test_raw_32_byte_seed() {
162        let seed_bytes = [42u8; 32];
163        let seed = parse_ed25519_seed(&seed_bytes).unwrap();
164        assert_eq!(seed.as_bytes(), &seed_bytes);
165    }
166
167    #[test]
168    fn test_octet_wrapped_34_bytes() {
169        let mut wrapped = vec![0x04, 0x20];
170        wrapped.extend_from_slice(&[7u8; 32]);
171        let seed = parse_ed25519_seed(&wrapped).unwrap();
172        assert_eq!(seed.as_bytes(), &[7u8; 32]);
173    }
174
175    #[test]
176    fn test_pkcs8_v2_roundtrip() {
177        let seed_bytes = [1u8; 32];
178        let pubkey_bytes = [2u8; 32];
179        let pkcs8 = build_ed25519_pkcs8_v2(&seed_bytes, &pubkey_bytes);
180        assert_eq!(pkcs8.len(), 85);
181
182        let (seed, maybe_pk) = parse_ed25519_key_material(&pkcs8).unwrap();
183        assert_eq!(seed.as_bytes(), &seed_bytes);
184        assert_eq!(maybe_pk.unwrap(), pubkey_bytes);
185    }
186

190        let pubkey_bytes = [4u8; 32];

192
193        let seed = parse_ed25519_seed(&pkcs8).unwrap();

203    #[test]
204    fn test_ring_83_byte_pkcs8_v2() {
205        // Ring produces 83-byte PKCS#8 v2 with implicit [1] tag for public key
206        let seed_bytes = [6u8; 32];
207        let pubkey_bytes = [7u8; 32];
208
209        let mut buf = Vec::with_capacity(83);
210        // SEQUENCE (81 bytes payload)
211        buf.extend_from_slice(&[0x30, 0x51]);
212        // INTEGER 1 (version = v2)
213        buf.extend_from_slice(&[0x02, 0x01, 0x01]);
214        // SEQUENCE { OID 1.3.101.112 }
215        buf.extend_from_slice(&[0x30, 0x05, 0x06, 0x03, 0x2b, 0x65, 0x70]);
216        // OCTET STRING { OCTET STRING { seed } }
217        buf.extend_from_slice(&[0x04, 0x22, 0x04, 0x20]);
218        buf.extend_from_slice(&seed_bytes);
219        // [1] IMPLICIT (33 bytes: 0x00 pad + pubkey)
220        buf.extend_from_slice(&[0x81, 0x21, 0x00]);
221        buf.extend_from_slice(&pubkey_bytes);
222        assert_eq!(buf.len(), 83);
223
224        let seed = parse_ed25519_seed(&buf).unwrap();
225        assert_eq!(seed.as_bytes(), &seed_bytes);
226
227        let (seed2, maybe_pk) = parse_ed25519_key_material(&buf).unwrap();
228        assert_eq!(seed2.as_bytes(), &seed_bytes);
229        assert_eq!(maybe_pk.unwrap(), pubkey_bytes);
230    }
