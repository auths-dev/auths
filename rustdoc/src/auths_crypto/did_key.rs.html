1//! DID:key encoding and decoding for Ed25519 public keys.
2//!
3//! Centralizes all `did:key` ↔ Ed25519 byte conversions in one place.
4//! The `did:key` method encodes a public key directly in the DID string
5//! using multicodec + base58btc, per the [did:key spec](https://w3c-ccg.github.io/did-method-key/).
6
7/// Ed25519 multicodec prefix (varint-encoded `0xED`).
8const ED25519_MULTICODEC: [u8; 2] = [0xED, 0x01];
9
10/// Errors from parsing or encoding `did:key` strings.
11#[derive(Debug, Clone, thiserror::Error, PartialEq, Eq)]
12#[non_exhaustive]
13pub enum DidKeyError {
14    #[error("DID must start with 'did:key:z', got: {0}")]
15    InvalidPrefix(String),
16
17    #[error("Base58 decoding failed: {0}")]
18    Base58DecodeFailed(String),
19
20    #[error("Unsupported or malformed multicodec: expected Ed25519 [0xED, 0x01]")]
21    UnsupportedMulticodec,
22
23    #[error("Invalid Ed25519 key length: expected 32 bytes, got {0}")]
24    InvalidKeyLength(usize),
25}
26
27/// Decode a `did:key:z...` string into a 32-byte Ed25519 public key.
28///
29/// Args:
30/// * `did`: A DID string in `did:key:z<base58btc>` format.
31///
32/// Usage:
33/// ```ignore
34/// let pk: [u8; 32] = did_key_to_ed25519("did:key:z6Mkf...")?;
35/// ```
36pub fn did_key_to_ed25519(did: &str) -> Result<[u8; 32], DidKeyError> {
37    let encoded = strip_did_key_prefix(did)?;
38    let decoded = decode_base58(encoded)?;
39    validate_multicodec_and_extract(&decoded)
40}
41
42/// Encode a 32-byte Ed25519 public key as a `did:key:z...` string.
43///
44/// Args:
45/// * `public_key`: A 32-byte Ed25519 public key.
46///
47/// Usage:
48/// ```ignore
49/// let did = ed25519_pubkey_to_did_key(&key_bytes);
50/// assert!(did.starts_with("did:key:z"));
51/// ```
52pub fn ed25519_pubkey_to_did_key(public_key: &[u8; 32]) -> String {
53    let mut prefixed = vec![0xED, 0x01];
54    prefixed.extend_from_slice(public_key);
55    let encoded = bs58::encode(prefixed).into_string();
56    format!("did:key:z{encoded}")
57}
58
59/// Encode a raw public key as a `did:keri:` string (base58-encoded).
60///
61/// Args:
62/// * `pk`: Raw public key bytes.
63pub fn ed25519_pubkey_to_did_keri(pk: &[u8]) -> String {
64    format!("did:keri:{}", bs58::encode(pk).into_string())
65}
66
67fn strip_did_key_prefix(did: &str) -> Result<&str, DidKeyError> {
68    did.strip_prefix("did:key:z")
69        .ok_or_else(|| DidKeyError::InvalidPrefix(did.to_string()))
70}
71
72fn decode_base58(encoded: &str) -> Result<Vec<u8>, DidKeyError> {
73    bs58::decode(encoded)
74        .into_vec()
75        .map_err(|e| DidKeyError::Base58DecodeFailed(e.to_string()))
76}
77
78fn validate_multicodec_and_extract(decoded: &[u8]) -> Result<[u8; 32], DidKeyError> {
79    if decoded.len() != 34
80        || decoded[0] != ED25519_MULTICODEC[0]
81        || decoded[1] != ED25519_MULTICODEC[1]
82    {
83        if decoded.len() != 34 {
84            return Err(DidKeyError::InvalidKeyLength(
85                decoded.len().saturating_sub(2),
86            ));
87        }
88        return Err(DidKeyError::UnsupportedMulticodec);
89    }
90
91    let mut key = [0u8; 32];
92    key.copy_from_slice(&decoded[2..]);
93    Ok(key)
94}

96#[cfg(test)]
97mod tests {
98    use super::*;
99
100    #[test]
101    fn roundtrip_encode_decode() {
102        let original = [42u8; 32];
103        let did = ed25519_pubkey_to_did_key(&original);
104        assert!(did.starts_with("did:key:z"));
105        let decoded = did_key_to_ed25519(&did).unwrap();
106        assert_eq!(decoded, original);
107    }
108
109    #[test]
110    fn rejects_invalid_prefix() {
111        let err = did_key_to_ed25519("did:web:example.com").unwrap_err();
112        assert!(matches!(err, DidKeyError::InvalidPrefix(_)));
113    }
114
115    #[test]
116    fn rejects_invalid_base58() {
117        let err = did_key_to_ed25519("did:key:z0OOO").unwrap_err();
118        assert!(matches!(err, DidKeyError::Base58DecodeFailed(_)));
119    }
