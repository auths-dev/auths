24/// Runtime dependency container for organization workflows.
25///
26/// Bundles all injected infrastructure adapters needed by org operations.
27/// The CLI constructs this from real implementations; tests inject fakes.
28///
29/// Args:
30/// * `registry`: Backend for reading/writing org member attestations.
31/// * `clock`: Wall-clock provider (use `SystemClock` in production, `MockClock` in tests).
32/// * `uuid_provider`: UUID generator for attestation resource IDs.
33/// * `signer`: Signing backend for creating cryptographic signatures.
34/// * `passphrase_provider`: Provider for obtaining key decryption passphrases.
35///
36/// Usage:
37/// ```ignore
38/// let ctx = OrgContext {
39///     registry: &backend,
40///     clock: &SystemClock,
41///     uuid_provider: &uuid_provider,
42///     signer: &signer,
43///     passphrase_provider: passphrase_provider.as_ref(),
44/// };
45/// let att = add_organization_member(&ctx, cmd)?;
46/// ```
47pub struct OrgContext<'a> {
48    /// Backend for reading/writing org member attestations.
49    pub registry: &'a dyn RegistryBackend,
50    /// Wall-clock provider (use `SystemClock` in production, `MockClock` in tests).
51    pub clock: &'a dyn ClockProvider,
52    /// UUID generator for attestation resource IDs.
53    pub uuid_provider: &'a dyn UuidProvider,
54    /// Signing backend for creating cryptographic signatures.
55    pub signer: &'a dyn SecureSigner,
56    /// Provider for obtaining key decryption passphrases.
57    pub passphrase_provider: &'a dyn PassphraseProvider,
58}

60/// Ordering key for org member display: admin < member < readonly < unknown.
61///
62/// Args:
63/// * `role`: Optional role as stored in an attestation.
64///
65/// Usage:
66/// ```ignore
67/// members.sort_by(|a, b| member_role_order(&a.role).cmp(&member_role_order(&b.role)));
68/// ```
69pub fn member_role_order(role: &Option<Role>) -> u8 {
70    match role {
71        Some(Role::Admin) => 0,
72        Some(Role::Member) => 1,
73        Some(Role::Readonly) => 2,
74        None => 3,
75    }
76}

78/// Find the first org-member attestation whose device public key matches `public_key_hex`
79/// and which holds the `manage_members` capability.
80///
81/// Args:
82/// * `backend`: Registry backend to query.
83/// * `org_prefix`: The KERI method-specific ID of the organization.
84/// * `public_key_hex`: Hex-encoded device public key of the candidate admin.
85///
86/// Usage:
87/// ```ignore
88/// let admin = find_admin(backend, "EOrg1234567890", &pubkey_hex)?;
89/// ```
90pub(crate) fn find_admin(
91    backend: &dyn RegistryBackend,
92    org_prefix: &str,
93    public_key_hex: &str,
94) -> Result<Attestation, OrgError> {
95    let signer_bytes = hex::decode(public_key_hex)
96        .map_err(|e| OrgError::InvalidPublicKey(format!("hex decode failed: {e}")))?;
97
98    let mut found: Option<Attestation> = None;
99
100    backend
101        .visit_org_member_attestations(org_prefix, &mut |entry| {
102            if let Ok(att) = &entry.attestation
103                && att.device_public_key.as_bytes().as_slice() == signer_bytes.as_slice()
104                && !att.is_revoked()
105                && att.capabilities.contains(&Capability::manage_members())
106            {
107                found = Some(att.clone());
108                return ControlFlow::Break(());
109            }
110            ControlFlow::Continue(())
111        })
112        .map_err(|e| OrgError::Storage(e.to_string()))?;
113
114    found.ok_or_else(|| OrgError::AdminNotFound {
115        org: org_prefix.to_owned(),
116    })
117}

119/// Find a member's current attestation by their DID within an org.
120///
121/// Args:
122/// * `backend`: Registry backend to query.
123/// * `org_prefix`: The KERI method-specific ID of the organization.
124/// * `member_did`: Full DID of the member to look up.
125///
126/// Usage:
127/// ```ignore
128/// let att = find_member(backend, "EOrg1234567890", "did:key:z6Mk...")?;
129/// ```
130pub(crate) fn find_member(
131    backend: &dyn RegistryBackend,
132    org_prefix: &str,
133    member_did: &str,
134) -> Result<Option<Attestation>, OrgError> {
135    let mut found: Option<Attestation> = None;
136
137    backend
138        .visit_org_member_attestations(org_prefix, &mut |entry| {
139            if entry.did.to_string() == member_did
140                && let Ok(att) = &entry.attestation
141            {
142                found = Some(att.clone());
143                return ControlFlow::Break(());
144            }
145            ControlFlow::Continue(())
146        })
147        .map_err(|e| OrgError::Storage(e.to_string()))?;
148
149    Ok(found)
150}

154fn parse_capabilities(raw: &[String]) -> Result<Vec<Capability>, OrgError> {
155    raw.iter()
156        .map(|s| {
157            Capability::try_from(s.clone()).map_err(|e| OrgError::InvalidCapability {
158                cap: s.clone(),
159                reason: e.to_string(),
160            })
161        })
162        .collect()
163}

261/// Add a new member to an organization with a cryptographically signed attestation.
262///
263/// Verifies that the signer holds the `manage_members` capability, creates a
264/// signed attestation via `create_signed_attestation` from auths-id, and stores
265/// the result in the registry backend.
266///
267/// Args:
268/// * `ctx`: Organization context with injected infrastructure adapters.
269/// * `cmd`: Add-member command with org prefix, member DID, role, and capabilities.
270///
271/// Usage:
272/// ```ignore
273/// let att = add_organization_member(&ctx, cmd)?;
274/// println!("Added member: {}", att.subject);
275/// ```
276pub fn add_organization_member(
277    ctx: &OrgContext,
278    cmd: AddMemberCommand,
279) -> Result<Attestation, OrgError> {
280    let admin_att = find_admin(ctx.registry, &cmd.org_prefix, &cmd.admin_public_key_hex)?;
281    let parsed_caps = parse_capabilities(&cmd.capabilities)?;
282    let now = ctx.clock.now();
283    let rid = ctx.uuid_provider.new_id().to_string();
284
285    let member_did = DeviceDID::new(&cmd.member_did);
286    let meta = AttestationMetadata {
287        note: cmd
288            .note
289            .or_else(|| Some(format!("Added as {} by {}", cmd.role, admin_att.subject))),
290        timestamp: Some(now),
291        expires_at: None,
292    };
293
294    let attestation = create_signed_attestation(
295        now,
296        &rid,
297        &admin_att.issuer,
298        &member_did,
299        cmd.member_public_key.as_bytes(),
300        Some(serde_json::json!({
301            "org_role": cmd.role.to_string(),
302            "org_did": format!("did:keri:{}", cmd.org_prefix),
303        })),
304        &meta,
305        ctx.signer,
306        ctx.passphrase_provider,
307        Some(&cmd.signer_alias),
308        None,
309        parsed_caps,
310        Some(cmd.role),
311        Some(IdentityDID::new(admin_att.subject.to_string())),
312    )
313    .map_err(|e| OrgError::Signing(e.to_string()))?;
314
315    ctx.registry
316        .store_org_member(&cmd.org_prefix, &attestation)
317        .map_err(|e| OrgError::Storage(e.to_string()))?;
318
319    Ok(attestation)
320}

322/// Revoke an existing org member with a cryptographically signed revocation.
323///
324/// Verifies that the signer holds `manage_members`, checks the member exists
325/// and is not already revoked, then creates a signed revocation attestation
326/// via `create_signed_revocation` from auths-id.
327///
328/// Args:
329/// * `ctx`: Organization context with injected infrastructure adapters.
330/// * `cmd`: Revoke-member command with org prefix and member DID.
331///
332/// Usage:
333/// ```ignore
334/// let revoked = revoke_organization_member(&ctx, cmd)?;
335/// assert!(revoked.is_revoked());
336/// ```
337pub fn revoke_organization_member(
338    ctx: &OrgContext,
339    cmd: RevokeMemberCommand,
340) -> Result<Attestation, OrgError> {
341    let admin_att = find_admin(ctx.registry, &cmd.org_prefix, &cmd.admin_public_key_hex)?;
342
343    let existing =
344        find_member(ctx.registry, &cmd.org_prefix, &cmd.member_did)?.ok_or_else(|| {
345            OrgError::MemberNotFound {
346                org: cmd.org_prefix.clone(),
347                did: cmd.member_did.clone(),
348            }
349        })?;
350
351    if existing.is_revoked() {
352        return Err(OrgError::AlreadyRevoked {
353            did: cmd.member_did.clone(),
354        });
355    }
356
357    let now = ctx.clock.now();
358    let member_did = DeviceDID::new(&cmd.member_did);
359
360    let revocation = create_signed_revocation(
361        admin_att.rid.as_str(),
362        &admin_att.issuer,
363        &member_did,
364        cmd.member_public_key.as_bytes(),
365        cmd.note,
366        None,
367        now,
368        ctx.signer,
369        ctx.passphrase_provider,
370        &cmd.signer_alias,
371    )
372    .map_err(|e| OrgError::Signing(e.to_string()))?;
373
374    ctx.registry
375        .store_org_member(&cmd.org_prefix, &revocation)
376        .map_err(|e| OrgError::Storage(e.to_string()))?;
377
378    Ok(revocation)
379}

381/// Update the capability set of an org member.
382///
383/// Verifies that the signer holds `manage_members`, checks the member exists
384/// and is not revoked, replaces their capability set, and re-stores.
385///
386/// Args:
387/// * `backend`: Registry backend for storage.
388/// * `clock`: Clock provider for the update timestamp.
389/// * `cmd`: Update-capabilities command with org prefix, member DID, and new capabilities.
390///
391/// Usage:
392/// ```ignore
393/// let updated = update_member_capabilities(backend, clock, cmd)?;
394/// ```
395pub fn update_member_capabilities(
396    backend: &dyn RegistryBackend,
397    clock: &dyn ClockProvider,
398    cmd: UpdateCapabilitiesCommand,
399) -> Result<Attestation, OrgError> {
400    find_admin(backend, &cmd.org_prefix, &cmd.public_key_hex)?;
401
402    let existing = find_member(backend, &cmd.org_prefix, &cmd.member_did)?.ok_or_else(|| {
403        OrgError::MemberNotFound {
404            org: cmd.org_prefix.clone(),
405            did: cmd.member_did.clone(),
406        }
407    })?;
408
409    if existing.is_revoked() {
410        return Err(OrgError::AlreadyRevoked {
411            did: cmd.member_did.clone(),
412        });
413    }
414
415    let parsed_caps = parse_capabilities(&cmd.capabilities)?;
416    let mut updated = existing;
417    updated.capabilities = parsed_caps;
418    updated.timestamp = Some(clock.now());
419
420    backend
421        .store_org_member(&cmd.org_prefix, &updated)
422        .map_err(|e| OrgError::Storage(e.to_string()))?;
423
424    Ok(updated)
