4//! 1. `compute_rotation_event` — pure, deterministic RotEvent construction.

33/// Computes a KERI rotation event and its canonical serialization.
34///
35/// Pure function — deterministic given fixed inputs. Signs the event bytes with
36/// `next_keypair` (the pre-committed future key becoming the new current key).
37/// `new_next_keypair` is the freshly generated key committed for the next rotation.
38///
39/// Args:
40/// * `state`: Current key state from the registry.
41/// * `next_keypair`: Pre-committed next key (becomes new current signer after rotation).
42/// * `new_next_keypair`: Freshly generated keypair committed for the next rotation.
43/// * `witness_config`: Optional witness configuration.
44///
45/// Returns `(event, canonical_bytes)` where `canonical_bytes` is the exact
46/// byte sequence to write to the KEL — do not re-serialize.
47///
48/// Usage:
49/// ```ignore
50/// let (rot, bytes) = compute_rotation_event(&state, &next_kp, &new_next_kp, None)?;
51/// ```
52pub fn compute_rotation_event(
53    state: &KeyState,
54    next_keypair: &Ed25519KeyPair,
55    new_next_keypair: &Ed25519KeyPair,
56    witness_config: Option<&WitnessConfig>,
57) -> Result<(RotEvent, Vec<u8>), RotationError> {
58    let prefix = &state.prefix;
59
60    let new_current_pub_encoded = format!(
61        "D{}",
62        URL_SAFE_NO_PAD.encode(next_keypair.public_key().as_ref())
63    );
64    let new_next_commitment = compute_next_commitment(new_next_keypair.public_key().as_ref());
65
66    let (bt, b) = match witness_config {
67        Some(cfg) if cfg.is_enabled() => (
68            cfg.threshold.to_string(),
69            cfg.witness_urls.iter().map(|u| u.to_string()).collect(),
70        ),
71        _ => ("0".to_string(), vec![]),
72    };
73
74    let new_sequence = state.sequence + 1;
75    let mut rot = RotEvent {
76        v: KERI_VERSION.to_string(),
77        d: Said::default(),
78        i: prefix.clone(),
79        s: KeriSequence::new(new_sequence),
80        p: state.last_event_said.clone(),
81        kt: "1".to_string(),
82        k: vec![new_current_pub_encoded],
83        nt: "1".to_string(),
84        n: vec![new_next_commitment],
85        bt,
86        b,
87        a: vec![],
88        x: String::new(),
89    };
90
91    let rot_json = serde_json::to_vec(&Event::Rot(rot.clone()))
92        .map_err(|e| RotationError::RotationFailed(format!("serialization failed: {e}")))?;
93    rot.d = compute_said(&rot_json);
94
95    let canonical = serialize_for_signing(&Event::Rot(rot.clone()))
96        .map_err(|e| RotationError::RotationFailed(format!("serialize for signing failed: {e}")))?;
97    let sig = next_keypair.sign(&canonical);
98    rot.x = URL_SAFE_NO_PAD.encode(sig.as_ref());
99
100    let event_bytes = serialize_for_signing(&Event::Rot(rot.clone()))
101        .map_err(|e| RotationError::RotationFailed(format!("final serialization failed: {e}")))?;
102
103    Ok((rot, event_bytes))
104}

106/// Key material required for the keychain side of `apply_rotation`.
107pub struct RotationKeyMaterial {
108    /// DID of the identity being rotated.
109    pub did: IdentityDID,
110    /// Alias to store the new current key (the former pre-committed next key).
111    pub next_alias: KeyAlias,
112    /// Alias for the future pre-committed key (committed in this rotation).
113    pub new_next_alias: KeyAlias,
114    /// Pre-committed next key alias to delete after successful rotation.
115    pub old_next_alias: KeyAlias,
116    /// Encrypted new current key bytes to store in the keychain.
117    pub new_current_encrypted: Vec<u8>,
118    /// Encrypted new next key bytes to store for future rotation.
119    pub new_next_encrypted: Vec<u8>,
120}

122/// Applies a computed rotation event to the registry and keychain.
123///
124/// Writes the KEL event first, then updates the keychain. If the KEL append
125/// succeeds but the subsequent keychain write fails, returns
126/// `RotationError::PartialRotation` so the caller can surface a recovery path.
127///
128/// # NOTE: non-atomic — KEL and keychain writes are not transactional.
129/// Recovery: re-run rotation with the same new key to replay the keychain write.
130///
131/// Args:
132/// * `rot`: The pre-computed rotation event to append to the KEL.
133/// * `prefix`: KERI identifier prefix (the `did:keri:` suffix).
134/// * `key_material`: Encrypted key material and aliases for keychain operations.
135/// * `registry`: Registry backend for KEL append.
136/// * `key_storage`: Keychain for storing rotated key material.
137///
138/// Usage:
139/// ```ignore
140/// apply_rotation(&rot, prefix, key_material, registry.as_ref(), key_storage.as_ref())?;
141/// ```
142pub fn apply_rotation(
143    rot: &RotEvent,
144    prefix: &Prefix,
145    key_material: RotationKeyMaterial,
146    registry: &(dyn RegistryBackend + Send + Sync),
147    key_storage: &(dyn KeyStorage + Send + Sync),
148) -> Result<(), RotationError> {
149    registry
150        .append_event(prefix, &Event::Rot(rot.clone()))
151        .map_err(|e| RotationError::RotationFailed(format!("KEL append failed: {e}")))?;
152
153    // NOTE: non-atomic — KEL and keychain writes are not transactional.
154    // If the keychain write fails here, the KEL is already ahead.
155    let keychain_result = (|| {
156        key_storage
157            .store_key(
158                &key_material.next_alias,
159                &key_material.did,
160                KeyRole::Primary,
161                &key_material.new_current_encrypted,
162            )
163            .map_err(|e| e.to_string())?;
164
165        key_storage
166            .store_key(
167                &key_material.new_next_alias,
168                &key_material.did,
169                KeyRole::NextRotation,
170                &key_material.new_next_encrypted,
171            )
172            .map_err(|e| e.to_string())?;
173
174        let _ = key_storage.delete_key(&key_material.old_next_alias);
175
176        Ok::<(), String>(())
177    })();
178
179    keychain_result.map_err(RotationError::PartialRotation)
180}

182/// Rotates the signing keys for an existing KERI identity.
183///
184/// Args:
185/// * `config` - Configuration for the rotation including aliases and paths.
186/// * `ctx` - The application context containing storage adapters.
187/// * `clock` - Provider for timestamps.
188///
189/// Usage:
190/// ```ignore
191/// let result = rotate_identity(
192///     IdentityRotationConfig {
193///         repo_path: PathBuf::from("/home/user/.auths"),
194///         identity_key_alias: Some("main".into()),
195///         next_key_alias: None,
196///     },
197///     &ctx,
198///     &SystemClock,
199/// )?;
200/// println!("Rotated to: {}...", result.new_key_fingerprint);
201/// ```
202pub fn rotate_identity(
203    config: IdentityRotationConfig,
204    ctx: &AuthsContext,
205    clock: &dyn ClockProvider,
206) -> Result<IdentityRotationResult, RotationError> {
207    let (identity, prefix, current_alias) = resolve_rotation_context(&config, ctx)?;
208    let next_alias = config.next_key_alias.unwrap_or_else(|| {
209        KeyAlias::new_unchecked(format!(
210            "{}-rotated-{}",
211            current_alias,
212            clock.now().format("%Y%m%d%H%M%S")
213        ))
214    });
215
216    let previous_key_fingerprint = extract_previous_fingerprint(ctx, &current_alias)?;
217
218    let state = ctx
219        .registry
220        .get_key_state(&prefix)
221        .map_err(|e| RotationError::KelHistoryFailed(e.to_string()))?;
222
223    let (decrypted_next_pkcs8, old_next_alias) =
224        retrieve_precommitted_key(&identity.controller_did, &current_alias, &state, ctx)?;
225
226    let (rot, new_next_pkcs8) = generate_rotation_keys(&identity, &state, &decrypted_next_pkcs8)?;
227
228    finalize_rotation_storage(
229        FinalizeParams {
230            did: &identity.controller_did,
231            prefix: &prefix,
232            next_alias: &next_alias,
233            old_next_alias: &old_next_alias,
234            current_pkcs8: &decrypted_next_pkcs8,
235            new_next_pkcs8: new_next_pkcs8.as_ref(),
236            rot: &rot,
237            state: &state,
238        },
239        ctx,
240    )?;
241
242    let (_, new_pubkey) = load_seed_and_pubkey(&decrypted_next_pkcs8)
243        .map_err(|e| RotationError::RotationFailed(e.to_string()))?;
244
245    Ok(IdentityRotationResult {
246        controller_did: identity.controller_did,
247        new_key_fingerprint: hex::encode(&new_pubkey[..8]),
248        previous_key_fingerprint,
249        sequence: state.sequence + 1,
250    })
251}

253/// Resolves the identity and determines which key alias is currently active.
254fn resolve_rotation_context(
255    config: &IdentityRotationConfig,
256    ctx: &AuthsContext,
257) -> Result<(ManagedIdentity, Prefix, KeyAlias), RotationError> {
258    let identity =
259        ctx.identity_storage
260            .load_identity()
261            .map_err(|_| RotationError::IdentityNotFound {
262                path: config.repo_path.clone(),
263            })?;
264
265    let prefix_str = identity
266        .controller_did
267        .as_str()
268        .strip_prefix("did:keri:")
269        .ok_or_else(|| {
270            RotationError::RotationFailed(format!(
271                "invalid DID format, expected 'did:keri:': {}",
272                identity.controller_did
273            ))
274        })?;
275    let prefix = Prefix::new_unchecked(prefix_str.to_string());
276
277    let current_alias = match &config.identity_key_alias {
278        Some(alias) => alias.clone(),
279        None => {
280            let aliases = ctx
281                .key_storage
282                .list_aliases_for_identity(&identity.controller_did)
283                .map_err(|e| RotationError::RotationFailed(format!("alias lookup failed: {e}")))?;
284            aliases
285                .into_iter()
286                .find(|a| !a.contains("--next-"))
287                .ok_or_else(|| {
288                    RotationError::KeyNotFound(format!(
289                        "no active signing key for {}",
290                        identity.controller_did
291                    ))
292                })?
293        }
294    };
295
296    Ok((identity, prefix, current_alias))
297}

299fn extract_previous_fingerprint(
300    ctx: &AuthsContext,
301    current_alias: &KeyAlias,
302) -> Result<String, RotationError> {
303    let old_pubkey_bytes = extract_public_key_bytes(
304        ctx.key_storage.as_ref(),
305        current_alias,
306        ctx.passphrase_provider.as_ref(),
307    )
308    .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
309
310    Ok(hex::encode(&old_pubkey_bytes[..8]))
311}

313/// Retrieves and decrypts the key that was committed in the previous KERI event.
314fn retrieve_precommitted_key(
315    did: &IdentityDID,
316    current_alias: &KeyAlias,
317    state: &KeyState,
318    ctx: &AuthsContext,
319) -> Result<(Zeroizing<Vec<u8>>, KeyAlias), RotationError> {
320    let target_alias =
321        KeyAlias::new_unchecked(format!("{}--next-{}", current_alias, state.sequence));
322
323    let (did_check, _role, encrypted_next) =
324        ctx.key_storage.load_key(&target_alias).map_err(|e| {
325            RotationError::KeyNotFound(format!(
326                "pre-committed next key '{}' not found: {e}",
327                target_alias
328            ))
329        })?;
330
331    if did != &did_check {
332        return Err(RotationError::RotationFailed(format!(
333            "DID mismatch for pre-committed key '{}': expected {}, found {}",
334            target_alias, did, did_check
335        )));
336    }
337
338    let pass = ctx
339        .passphrase_provider
340        .get_passphrase(&format!(
341            "Enter passphrase for pre-committed key '{}':",
342            target_alias
343        ))
344        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
345
346    let decrypted = decrypt_keypair(&encrypted_next, &pass)
347        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
348
349    let keypair = load_keypair_from_der_or_seed(&decrypted)
350        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
351
352    if !verify_commitment(keypair.public_key().as_ref(), &state.next_commitment[0]) {
353        return Err(RotationError::RotationFailed(
354            "commitment mismatch: next key does not match previous commitment".into(),
355        ));
356    }
357
358    Ok((decrypted, target_alias))
359}

361/// Generates the new rotation event and the next forward-looking key commitment.
362fn generate_rotation_keys(
363    identity: &ManagedIdentity,
364    state: &KeyState,
365    current_key_pkcs8: &[u8],
366) -> Result<(RotEvent, ring::pkcs8::Document), RotationError> {
367    let witness_config: Option<WitnessConfig> = identity
368        .metadata
369        .as_ref()
370        .and_then(|m| m.get("witness_config"))
371        .and_then(|wc| serde_json::from_value(wc.clone()).ok());
372
373    let rng = SystemRandom::new();
374    let new_next_pkcs8 = Ed25519KeyPair::generate_pkcs8(&rng)
375        .map_err(|e| RotationError::RotationFailed(format!("key generation failed: {e}")))?;
376    let new_next_keypair = Ed25519KeyPair::from_pkcs8(new_next_pkcs8.as_ref())
377        .map_err(|e| RotationError::RotationFailed(format!("key construction failed: {e}")))?;
378
379    let next_keypair = load_keypair_from_der_or_seed(current_key_pkcs8)
380        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
381
382    let (rot, _event_bytes) = compute_rotation_event(
383        state,
384        &next_keypair,
385        &new_next_keypair,
386        witness_config.as_ref(),
387    )?;
388
389    Ok((rot, new_next_pkcs8))
390}

392struct FinalizeParams<'a> {
393    did: &'a IdentityDID,
394    prefix: &'a Prefix,
395    next_alias: &'a KeyAlias,
396    old_next_alias: &'a KeyAlias,
397    current_pkcs8: &'a [u8],
398    new_next_pkcs8: &'a [u8],
399    rot: &'a RotEvent,
400    state: &'a KeyState,
401}
402
403/// Encrypts and persists the new current and next keys to secure storage.
404fn finalize_rotation_storage(
405    params: FinalizeParams<'_>,
406    ctx: &AuthsContext,
407) -> Result<(), RotationError> {
408    let new_pass = ctx
409        .passphrase_provider
410        .get_passphrase(&format!(
411            "Create passphrase for new key alias '{}':",
412            params.next_alias
413        ))
414        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
415
416    let confirm_pass = ctx
417        .passphrase_provider
418        .get_passphrase(&format!("Confirm passphrase for '{}':", params.next_alias))
419        .map_err(|e| RotationError::KeyDecryptionFailed(e.to_string()))?;
420
421    if new_pass != confirm_pass {
422        return Err(RotationError::RotationFailed(format!(
423            "passphrases do not match for alias '{}'",
424            params.next_alias
425        )));
426    }
427
428    let encrypted_new_current = encrypt_keypair(params.current_pkcs8, &new_pass)
429        .map_err(|e| RotationError::RotationFailed(format!("encrypt new current key: {e}")))?;
430
431    let new_next_seed = extract_seed_bytes(params.new_next_pkcs8)
432        .map_err(|e| RotationError::RotationFailed(format!("extract new next seed: {e}")))?;
433    let new_next_seed_pkcs8 = encode_seed_as_pkcs8(new_next_seed)
434        .map_err(|e| RotationError::RotationFailed(format!("encode new next seed: {e}")))?;
435    let encrypted_new_next = encrypt_keypair(&new_next_seed_pkcs8, &new_pass)
436        .map_err(|e| RotationError::RotationFailed(format!("encrypt new next key: {e}")))?;
437
438    let new_sequence = params.state.sequence + 1;
439    let new_next_alias =
440        KeyAlias::new_unchecked(format!("{}--next-{}", params.next_alias, new_sequence));
441
442    let key_material = RotationKeyMaterial {
443        did: params.did.clone(),
444        next_alias: params.next_alias.clone(),
445        new_next_alias,
446        old_next_alias: params.old_next_alias.clone(),
447        new_current_encrypted: encrypted_new_current.to_vec(),
448        new_next_encrypted: encrypted_new_next.to_vec(),
449    };
450
451    apply_rotation(
452        params.rot,
453        params.prefix,
454        key_material,
455        ctx.registry.as_ref(),
456        ctx.key_storage.as_ref(),
457    )
458}
