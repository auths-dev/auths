1//! Git allowed-signers file computation from device attestations.
2
3use auths_id::error::StorageError;
4use auths_id::storage::attestation::AttestationSource;
5use auths_verifier::core::Attestation;
6use ssh_key::PublicKey as SshPublicKey;
7use ssh_key::public::Ed25519PublicKey;
8use thiserror::Error;
9
10/// A single entry in a Git allowed_signers file.
11#[derive(Debug, Clone)]
12pub struct AllowedSignerEntry {
13    /// Email or principal used by Git to identify the signer.
14    pub principal: String,
15    /// OpenSSH public key string (e.g. `"ssh-ed25519 AAAA..."`).
16    pub ssh_public_key: String,
17}
18
19/// Errors that can occur during Git integration operations.
20#[derive(Debug, Error)]
21pub enum GitIntegrationError {
22    /// Attestation storage could not be read.
23    #[error("failed to load attestations: {0}")]
24    Storage(#[from] StorageError),
25    /// Raw public key bytes have an unexpected length.
26    #[error("invalid Ed25519 public key length: expected 32, got {0}")]
27    InvalidKeyLength(usize),
28    /// SSH key encoding failed.
29    #[error("failed to encode SSH public key: {0}")]
30    SshKeyEncoding(String),
31}
32
33/// Compute the list of allowed-signer entries from an attestation source.
34///
35/// Skips revoked attestations and devices whose public key cannot be
36/// parsed; those are silently dropped.
37///
38/// Args:
39/// * `source`: Attestation storage backend.
40///
41/// Usage:
42/// ```ignore
43/// let entries = generate_allowed_signers(&storage)?;
44/// let file_content = format_allowed_signers_file(&entries);
45/// ```
46pub fn generate_allowed_signers(
47    source: &dyn AttestationSource,
48) -> Result<Vec<AllowedSignerEntry>, GitIntegrationError> {
49    let attestations = source.load_all_attestations()?;
50    let mut entries: Vec<AllowedSignerEntry> = attestations
51        .iter()
52        .filter(|att| !att.is_revoked())
53        .filter_map(|att| {
54            let ssh_key = public_key_to_ssh(att.device_public_key.as_bytes()).ok()?;
55            Some(AllowedSignerEntry {
56                principal: principal_for(att),
57                ssh_public_key: ssh_key,
58            })
59        })
60        .collect();
61    entries.sort_by(|a, b| a.principal.cmp(&b.principal));
62    entries.dedup_by(|a, b| a.principal == b.principal && a.ssh_public_key == b.ssh_public_key);
63    Ok(entries)
64}
65
66/// Format a list of `AllowedSignerEntry` values as an `allowed_signers` file.
67///
68/// Each line has the form `<principal> namespaces="git" <ssh-key>`.
69/// Returns an empty string when `entries` is empty, otherwise the file
70/// ends with a trailing newline.
71///
72/// Args:
73/// * `entries`: Computed allowed-signer entries.
74///
75/// Usage:
76/// ```ignore
77/// let content = format_allowed_signers_file(&entries);
78/// std::fs::write(path, content)?;
79/// ```
80pub fn format_allowed_signers_file(entries: &[AllowedSignerEntry]) -> String {
81    if entries.is_empty() {
82        return String::new();
83    }
84    let lines: Vec<String> = entries
85        .iter()
86        .map(|e| format!("{} namespaces=\"git\" {}", e.principal, e.ssh_public_key))
87        .collect();
88    format!("{}\n", lines.join("\n"))
89}

91/// Convert raw Ed25519 public key bytes to an OpenSSH public key string.
92///
93/// Args:
94/// * `public_key_bytes`: 32-byte Ed25519 public key.
95///
96/// Usage:
97/// ```ignore
98/// let openssh = public_key_to_ssh(&bytes)?;
99/// ```
100pub fn public_key_to_ssh(public_key_bytes: &[u8]) -> Result<String, GitIntegrationError> {
101    if public_key_bytes.len() != 32 {
102        return Err(GitIntegrationError::InvalidKeyLength(
103            public_key_bytes.len(),
104        ));
105    }
106    let ed25519_pk = Ed25519PublicKey::try_from(public_key_bytes)
107        .map_err(|e| GitIntegrationError::SshKeyEncoding(e.to_string()))?;
108    let ssh_pk = SshPublicKey::from(ed25519_pk);
109    ssh_pk
110        .to_openssh()
111        .map_err(|e| GitIntegrationError::SshKeyEncoding(e.to_string()))
112}

114fn principal_for(att: &Attestation) -> String {
115    if let Some(ref payload) = att.payload
116        && let Some(email) = payload.get("email").and_then(|v| v.as_str())
117        && !email.is_empty()
118    {
119        return email.to_string();
120    }
121    let did_str = att.subject.to_string();
122    let local_part = did_str.strip_prefix("did:key:").unwrap_or(&did_str);
123    format!("{}@auths.local", local_part)
124}
